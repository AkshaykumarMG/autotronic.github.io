src/app/utilities/interceptor/error.interceptor.ts

errorInterceptor
Type : HttpInterceptorFn
Default value : (req, next) => { // inject dependencies const router = inject(Router); const authService = inject(AuthService); const localStorageService = inject(LocalStorageService); // intercept requests and handle errors return next(req).pipe( // tap to handle response events, including errors tap(async (event: HttpEvent<any>) => { // handle response events only if it's an instance of HttpResponse if(event instanceof HttpResponse) { // check if error is related to http request if (event instanceof HttpErrorResponse) { // check if user is unauthorized (401) and not logged in if (event.status === 401) { if(authService.isLoggedIn()) { localStorageService.remove('user'); localStorageService.remove('store'); localStorageService.remove('ne'); localStorageService.remove('user'); await authService.OnLogout(); } // detail page is dynamic url so check by string if('/detail'.indexOf(router.routerState.snapshot.url)) return // check static pre defined url where login is not required if(['/prequalify-loan?isLoginSetup=true', '/prequalify-loan', '/', '/search', '/detail'].includes(router.routerState.snapshot.url)) return // redirect to login router.navigate(['/login']); } else if(event.status === 307) { router.navigate(['/user/store']); } else if(event.status === 0) { router.navigateByUrl('/server-down'); } else { return; } } } }) ); }
