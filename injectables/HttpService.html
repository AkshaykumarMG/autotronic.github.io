server_url

import { HttpClient, HttpErrorResponse, HttpHeaders } from '@angular/common/http';
import { Inject, Injectable, PLATFORM_ID } from '@angular/core';
import { catchError, lastValueFrom, throwError } from 'rxjs';
import { environment } from '../../../environments/environment';

@Injectable({
  providedIn: 'root'
})
export class HttpService {

  // Base server URL, can be overridden in the environment file.
  server_url = environment.backend_url;

  constructor(
    private httpService: HttpClient,
  ) { }

  /**
   * Performs an HTTP GET request to the server.
   *
   * This method sends a GET request to the server, appending the provided URL to the base server URL.
   * It includes credentials with the request and uses lastValueFrom to convert the Observable to a Promise.
   *
   * @param url - The endpoint URL to be appended to the base server URL for the GET request.
   * @returns A Promise that resolves with the server's response data if the request is successful.
   * @throws Will throw an error if the HTTP request fails or if there's a network issue.
   */
  async get(url: string) {
    return await lastValueFrom(this.httpService.get(this.server_url + url, { withCredentials: true }));
  }

  /**
   * Performs an HTTP GET request and returns an Observable.
   * 
   * This method sends a GET request to the server, appending the provided URL to the base server URL.
   * It handles any errors that may occur using RxJS operators.
   *
   * @param url - The endpoint URL to be appended to the base server URL for the GET request.
   * @returns An Observable that emits the server's response data if the request is successful.
   *          If an error occurs, the Observable will emit a user-friendly error message.
   */
  getRxJS(url: string) {
    return this.httpService.get(this.server_url + url).pipe(
      catchError(this.handleError)
    );
  }

  /**
   * Performs an HTTP GET request to an external URL.
   *
   * This method sends a GET request to the specified external URL using the HttpClient service.
   * It uses lastValueFrom to convert the Observable returned by HttpClient into a Promise,
   * making it easier to use with async/await.
   *
   * @param url - The full external URL to which the GET request will be sent.
   * @returns A Promise that resolves with the response data from the external URL.
   * @throws Will throw an error if the HTTP request fails or if there's a network issue.
   */
  async getWithOutsideURL(url: string) {
    return await lastValueFrom(this.httpService.get(url));
  }

  /**
   * Performs an HTTP GET request to an external URL and retrieves CSV data as text.
   *
   * This method sends a GET request to the specified external URL and expects the response
   * to be in CSV format. It sets the response type to 'text' to receive the raw CSV data.
   * The method uses lastValueFrom to convert the Observable returned by HttpClient into a Promise,
   * making it easier to use with async/await.
   *
   * @param url - The full external URL from which to retrieve the CSV data.
   * @returns A Promise that resolves with the CSV data as a string if the request is successful.
   *          If an error occurs, the promise will be rejected.
   * @throws Will throw an error if the HTTP request fails or if there's a network issue.
   */
  async getWithOutsideURLCSV(url: string) {
    return await lastValueFrom(this.httpService.get(url, { responseType: 'text' }));
  }

  /**
   * Performs an HTTP POST request to the server.
   *
   * This method sends a POST request to the server, appending the provided URL to the base server URL.
   * It includes the given data in the request body and sends credentials with the request.
   * Any errors that occur during the request are handled by the `handleError` method.
   *
   * @param url - The endpoint URL to be appended to the base server URL for the POST request.
   * @param data - An object containing the data to be sent in the request body.
   * @returns A Promise that resolves with the server's response data if the request is successful.
   *          If an error occurs, the promise will be rejected with a user-friendly error message.
   * @throws Will throw an error if the HTTP request fails or if there's a network issue.
   */
  async post(url: string, data: object) {
    return await lastValueFrom(this.httpService.post(this.server_url + url, data, { withCredentials: true }).pipe(
      catchError(this.handleError)
    ));
  }

  /**
   * Performs an HTTP POST request and returns an Observable.
   * 
   * This method sends a POST request to the server, appending the provided URL to the base server URL.
   * It includes the given data in the request body and handles any errors that may occur using RxJS operators.
   *
   * @param url - The endpoint URL to be appended to the base server URL for the POST request.
   * @param data - The data to be sent in the request body. Can be of any type.
   * @returns An Observable that emits the server's response data if the request is successful.
   *          If an error occurs, the Observable will emit a user-friendly error message.
   */
  postRxJS(url: string, data: any) {
    return this.httpService.post(this.server_url + url, data).pipe(
      catchError(this.handleError)
    );
  }

  /**
   * Performs a POST HTTP request to an external URL with JSON data.
   *
   * This method sends a POST request to the specified external URL, including the given data
   * in the request body as JSON. It uses lastValueFrom to convert the Observable returned by
   * HttpClient into a Promise, making it easier to use with async/await. Any errors that occur
   * during the request are handled by the `handleError` method.
   *
   * @param url - The full external URL for the POST request.
   * @param data - An object containing the data to be sent in the request body as JSON.
   * @returns A Promise that resolves with the server's response data if the request is successful.
   *          If an error occurs, the promise will be rejected with a user-friendly error message.
   * @throws Will throw an error if the HTTP request fails or if there's a network issue.
   */
  async postWithOutsideURL(url: string, data: object) {
    return await lastValueFrom(this.httpService.post(url, data, { headers: { 'Content-Type': 'application/json' } }).pipe(
      catchError(this.handleError)
    ));
  }

  /**
   * Performs a PUT HTTP request to the specified URL with custom headers.
   *
   * This method sends a PUT request to the provided URL, including the given data in the request body
   * and setting the Content-Type header based on the specified type. It uses lastValueFrom to convert
   * the Observable returned by HttpClient into a Promise, making it easier to use with async/await.
   * Any errors that occur during the request are handled by the `handleError` method.
   *
   * @param url - The full URL for the PUT request.
   * @param data - The data to be sent in the request body. Can be of any type.
   * @param type - The Content-Type header value to be set for the request.
   * @returns A Promise that resolves with the server's response data if the request is successful.
   *          If an error occurs, the promise will be rejected with a user-friendly error message.
   * @throws Will throw an error if the HTTP request fails or if there's a network issue.
   */
  async putURL(url: string, data: any, type: string) {
    const headers = new HttpHeaders({
      'Content-Type': type,
    });
    return await lastValueFrom(this.httpService.put(url, data, { ...headers }).pipe(
      catchError(this.handleError),
    ));
  }

  /**
   * Performs a PATCH HTTP request to the specified URL.
   *
   * This method sends a PATCH request to the server, appending the provided URL to the base server URL.
   * It includes the given data in the request body and sends credentials with the request.
   * Any errors that occur during the request are handled by the `handleError` method.
   *
   * @param url - The endpoint URL to be appended to the base server URL for the PATCH request.
   * @param data - The data to be sent in the request body. Can be of any type.
   * @returns A Promise that resolves with the server's response data if the request is successful.
   *          If an error occurs, the promise will be rejected with a user-friendly error message.
   * @throws Will throw an error if the HTTP request fails or if there's a network issue.
   */
  async patch(url: string, data: any) {
    return await lastValueFrom(this.httpService.patch(this.server_url + url, data, { withCredentials: true }).pipe(
      catchError(this.handleError)
    ));
  }

  /**
   * Performs a DELETE HTTP request to the specified URL.
   *
   * This method sends a DELETE request to the server, appending the provided URL to the base server URL.
   * It includes credentials with the request and handles any errors that may occur.
   *
   * @param url - The endpoint URL to be appended to the base server URL for the DELETE request.
   * @returns A Promise that resolves with the server's response data if the request is successful.
   *          If an error occurs, the promise will be rejected with a user-friendly error message.
   * @throws Will throw an error if the HTTP request fails or if there's a network issue.
   */
  async delete(url: string) {
    return await lastValueFrom(this.httpService.delete(this.server_url + url, { withCredentials: true }).pipe(
      catchError(this.handleError)
    ));
  }

  /**
   * Handles HTTP errors and provides a user-friendly error message.
   * 
   * This function processes HttpErrorResponse objects, logs appropriate error messages,
   * and returns an Observable that emits a user-facing error message.
   *
   * @param error - The HttpErrorResponse object containing error details.
   * @returns An Observable that emits a string containing a user-friendly error message.
   */
  private handleError(error: HttpErrorResponse) {
    if (error.status === 0) {
      // A client-side or network error occurred. Handle it accordingly.
      console.error('An error occurred:', error.error);
    } else {
      // The backend returned an unsuccessful response code.
      // The response body may contain clues as to what went wrong.
      console.error(
        `Backend returned code ${error.status}, body was: `, error.error);
    }
    const message = Array.isArray(error.error.message) ? error.error.message[0] : error.error.message;
    // Return an observable with a user-facing error message.
    return throwError(() => message);
  }

}
