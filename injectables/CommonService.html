import { Injectable } from '@angular/core';

@Injectable({
  providedIn: 'root'
})
export class CommonService {

  constructor() { }

  /**
   * Converts a string to title case by capitalizing the first letter of each word and replacing underscores with spaces.
   * 
   * @param key - The input string to be converted to title case.
   *              Can be any type, but non-string inputs may produce unexpected results.
   * 
   * @returns The converted string in title case format.
   *          Returns undefined if the input is falsy.
   *          Returns the original string if it consists entirely of uppercase letters.
   */
  replaceStringToCapitalize(key: any) {
    if (!key) return
    if (/^[A-Z]+$/.test(key)) return key;
    return key.toLowerCase().replace(/_/g, ' ').replace(/(?: |\b)(\w)/g, (key: string, p1: string) => {
      return key.toUpperCase();
    })
  }

  /**
   * Converts a string to lowercase, replacing spaces with underscores.
   * 
   * @param key - The input string to be converted to lowercase with underscores.
   *              Can be any type, but non-string inputs may produce unexpected results.
   * 
   * @returns The converted string in lowercase with underscores.
   *          Returns undefined if the input is falsy.
   */
  replaceStringToLowercase(key: any) {
    if (!key) return
    return key.toLowerCase().replace(/ /g, '_').replace(/(?: |\b)(\w)/g, (key: string, p1: string) => {
      return key.toLowerCase();
    })
  }

  /**
   * Sorts an array of objects based on a specified field.
   * 
   * @param data - The array of objects to be sorted.
   * @param field - The name of the field to sort by.
   * @param type - The type of the field to sort by. Can be 'string' for string comparison or any other value for numeric comparison.
   * @returns A new sorted array. If type is 'string', it sorts alphabetically (case-insensitive). Otherwise, it sorts numerically.
   */
  sortArrayField(data: any, field: string, type: string) {
    if (type === 'string') {
      return data.sort((a: any, b: any) => a[field].toLowerCase() > b[field].toLowerCase() ? 1 : -1);
    }
    return data.sort((a: any, b: any) => a[field] - b[field]);
  }

  /**
   * Converts an array of objects to a comma-separated string of values from a specified field.
   *
   * @param data - An array of objects from which to extract field values.
   * @param field - The name of the field whose values should be extracted and joined.
   * @returns A string containing the values of the specified field from all objects in the array, joined by commas.
   */
  setArrayToString(data: Array<any>, field: string): string {
    return data.map((element: any) => element[field]).join(', ');
  }

  /**
   * Generates a new unique object ID.
   * 
   * This function creates a 16-character hexadecimal string that can be used as a unique identifier.
   * The ID is composed of:
   * - A timestamp (first 8 characters): represents the current time in seconds since the Unix epoch, in hexadecimal.
   * - A random string (last 6 characters): generated using random hexadecimal digits.
   * 
   * @returns A 16-character lowercase hexadecimal string that can be used as a unique object identifier.
   */
  newObjectId() {
    const timestamp = Math.floor(new Date().getTime() / 1000).toString(16);
    const objectId = timestamp + 'xxxxxx'.replace(/[x]/g, () => {
      return Math.floor(Math.random() * 16).toString(16);
    }).toLowerCase();
  
    return objectId;
  }

  /**
   * Converts CSV formatted data to JSON.
   * 
   * This function takes a CSV string and converts it into an array of JavaScript objects.
   * The first line of the CSV is assumed to be the header, which defines the property names for the resulting objects.
   * 
   * @param data - The CSV data as a string. The first line should contain the headers.
   * @param delimiter - The character used to separate values in the CSV. Defaults to ','.
   * @returns An array of objects where each object represents a row from the CSV,
   *          with properties named according to the CSV headers.
   */
  CSVToJSON(data: any, delimiter = ',') {
    const titles = data.slice(0, data.indexOf('\n')).split(delimiter);
    return data
      .slice(data.indexOf('\n') + 1)
      .split('\n')
      .map((v: any) => {
        const values = v.split(delimiter);
        return titles.reduce(
          (obj: any, title: any, index: any) => ((obj[title] = values[index]), obj),
          {}
        );
      });
  };
}
