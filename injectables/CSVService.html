import { Injectable } from '@angular/core';

@Injectable({
  providedIn: 'root'
})
export class CSVService {

  constructor() { }

  downloadFile(data: any, header: Array<string>, field: Array<string>, filename = 'data') {
    // const arrHeader = ["name", "age", "country", "phone"];
    const csvData = this.ConvertToCSV(data, header, field); // arrHeader - argumanet
    // console.log(csvData);
    // return
    const blob = new Blob(['\ufeff' + csvData], { type: 'text/csv;charset=utf-8;' });
    const dwldLink = document.createElement("a");
    const url = URL.createObjectURL(blob);
    const isSafariBrowser = navigator.userAgent.indexOf('Safari') != -1 && navigator.userAgent.indexOf('Chrome') == -1;
    if (isSafariBrowser) {  //if Safari open in new window to save file with random filename.
      dwldLink.setAttribute("target", "_blank");
    }
    dwldLink.setAttribute("href", url);
    dwldLink.setAttribute("download", filename);
    dwldLink.style.visibility = "hidden";
    document.body.appendChild(dwldLink);
    dwldLink.click();
    document.body.removeChild(dwldLink);
  }

  /**
   * Converts a CSV string to a JSON object.
   *
   * This function takes a CSV formatted string and converts it into an array of JSON objects.
   * The first line of the CSV is assumed to be the header row, which is used to create the
   * property names for each object.
   *
   * @param data - The CSV formatted string to be converted.
   * @param delimiter - The character used to separate values in the CSV. Defaults to ','.
   * @returns An array of objects, where each object represents a row from the CSV,
   *          with properties named according to the CSV headers.
   */
  CSVToJSON(data: any, delimiter = ','): any[] {
    const titles = data.slice(0, data.indexOf('\n')).split(delimiter);
    return data
      .slice(data.indexOf('\n') + 1)
      .split('\n')
      .map((v: any) => {
        const values = v.split(delimiter);
        return titles.reduce(
          (obj: any, title: any, index: any) => ((obj[title] = values[index]), obj),
          {}
        );
      });
  };

  /**
   * Converts an array of objects into a CSV string format.
   * 
   * This function takes an array of objects and converts it into a CSV (Comma-Separated Values) 
   * string. It uses the provided header list for the first row and extracts values from the 
   * objects based on the field list.
   *
   * @param objData - The data to be converted. Can be either a JSON string or an array of objects.
   * @param headerList - An array of strings representing the headers for the CSV.
   * @param fieldList - An array of strings representing the object keys to be used for data extraction.
   * @returns A string in CSV format, with headers as the first row followed by data rows.
   */
  ConvertToCSV(objData: any, headerList: Array<string>, fieldList: Array<string>) {
    const array = typeof objData != 'object' ? JSON.parse(objData) : objData;
    let str = '';
    let row = '';
  
    for (const index in headerList) {
      row += `"${headerList[index]}",`;
    }
    row = row.slice(0, -1);
    str += `${row}\r\n`;
    for (const element of array) {
      let line = '';
      for (const index in fieldList) {
        const head = fieldList[index];
  
        line += line === '' ? this.strRep(element[head]) : `,${this.strRep(element[head])}`;
      }
      str += `${line}\r\n`;
    }
    return str;
  }

  /**
   * Converts various data types to string representations suitable for CSV format.
   * 
   * This function handles different data types and formats them appropriately for CSV output:
   * - Undefined or null values are converted to "N/A"
   * - Strings are wrapped in quotes
   * - Arrays of objects are processed based on specific properties (url, label, or header)
   * - Simple arrays are joined with commas and wrapped in quotes
   * - Numbers are converted to strings and wrapped in quotes
   * - Booleans are converted to "Active" or "Inactive"
   * - Other types are returned as-is
   * 
   * @param data - The input data to be converted. Can be a string, number, undefined, or object (including arrays)
   * @returns A string representation of the input data, formatted for CSV output
   */
  strRep(data: string | number | undefined | object) {
    if (typeof data == "undefined") {
      return "N/A";
    } else if (data === null) {
      return "N/A";
    } else if (typeof data == "string") {
      return `"${data}"`;
    } else if (typeof data == "object" && Array.isArray(data) && typeof data[0] === "object") {
      let line = '';
      if(data[0].url) {
        line = data.map((obj: any) =>  { return obj.url }).join(',');
      } else if(data[0].label) {
        line = data.map((obj: any) =>  { return obj.label }).join(',');
      } else {
        line = data.map((obj: any) =>  { return obj.header }).join(',');
      }
      return `"${line}"`;
    }
    else if (Array.isArray(data)) {
      return `"${data.join(', ')}"`;
    }
    else if (typeof data == "number") {
      return `"${data}"`;
    }
    else if (typeof data == "boolean") {
      return data ? "Active" : "Inactive";
    }
    else {
      return data;
    }
  }

}
