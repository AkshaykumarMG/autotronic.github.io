import { Inject, Injectable, InjectionToken } from '@angular/core';

export const BROWSER_STORAGE = new InjectionToken<Storage>('Browser Storage', {
  providedIn: 'root',
  factory: () => localStorage
});

@Injectable({
  providedIn: 'root'
})
export class LocalStorageService {

  constructor(
    @Inject(BROWSER_STORAGE) public storage: Storage
  ) { }

  /**
   * set key - value in local storage
   * @param key 
   * @param value 
   */
  set(key: string, value: string) {
    if (this.storage) {
      this.storage.setItem(key, value);
    }
  }

  /**
   * get value from loca storage by key
   * @param key 
   * @returns 
   */
  get(key: string) {
    if (this.storage) {
      return this.storage.getItem(key);
    }
    return null;
  }

  /**
   * set key - value by object 
   * @param key 
   * @param value 
   */
  setObject(key: string, value: Object) {
    if (this.storage) {
      this.storage.setItem(key, JSON.stringify(value));
    }
  }

  /**
   * get key value by object
   * @param key 
   * @returns 
   */
  getObject(key: string) {
    if (this.storage) {
      return JSON.parse(this.storage.getItem(key)!);
    }
  }

  /**
   * remove key from local storage
   * @param key 
   */
  remove(key: string) {
    if (this.storage) {
      this.storage.removeItem(key);
    }
  }

  /**
   * clear local storage
   */
  clear() {
    if (this.storage) {
      this.storage.clear();
    }
  }

  /**
 * Checks if the current platform is a browser.
 *
 * This method uses the `isPlatformBrowser` function from `@angular/common` to determine if the current
 * platform is a browser. The result is cached in the `storage` property for future use.
 *
 * @returns {boolean} Returns `true` if the current platform is a browser, `false` otherwise.
 */
  isPlateformBrowser() {
    return this.storage;
  }

  /**
   * Determines if the current user is a dealer.
   * 
   * This method checks the user object stored in local storage to determine
   * if the user's type is 'dlr' (dealer).
   * 
   * @returns {boolean | null} Returns true if the user is a dealer, false if not,
   * or null if storage is not available or the user object doesn't exist.
   */
  userIsDealer(): boolean | null {
    if (this.storage) {
      return this.getObject('user').type === 'dlr'
    }
    return null
  }
}
