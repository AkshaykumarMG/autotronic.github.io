export interface CarForm {
    name: string,
    value: string,
    data: Array<CommonElement>
}

export interface CommonElement {
    type: element,
    name: string,
    datatype: string, // type like number, text
    placeholder: string,
    value: any, // set default value
    format: textBoxtype, // format for textbox only
    readonly: boolean, // make element readonly
    data: LabelValueData[],
    search: LabelValueData[], // store auto complete - multi select search result
    accept: string, // file format extentions
    class: string,
    multiple: boolean, // file selection,
    mode: string, // media mode - add or delete
    required: boolean,
    error: string, // set error for element
    pattern?: { type: string, label: string, value: string }, // for specific textbox format regex
    dateFormat: string, // date format
    minDate: Date, // min date
    keyFilter: string, // data enter format with primeng int, alphanum
    maxSize: number, // file maxsize
    note: string, // note for specific element
    group_right: string, // input group element
    group_left: string, // input group element
    group_string: string, // input group element
    dependant: boolean, // check dependant on any element
    dependant_field: string, // get dependant field if dependant,
    onChange: boolean, // modify toggle as per dependant validation,
    changeField: string, // set disable field for main element,
    element: any // multiple selection field
    removed: boolean, // remove element from selection form
}

export interface LabelValueData {
    label: string,
    value: string
}
