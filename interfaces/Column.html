import { Component, OnInit, ViewEncapsulation } from '@angular/core';
import { Router } from '@angular/router';

// date format
import dayjs from 'dayjs';

// custom service
import { HttpService } from '../../../utilities/services/http.service';
import { CommonService } from '../../../utilities/services/common.service';
import { LocalStorageService } from '../../../utilities/services/local-storage.service';
import { CSVService } from '../../../utilities/services/csv.service';

// primeng
import { ConfirmationService, MessageService } from 'primeng/api';
import { TablePageEvent } from 'primeng/table';
import { InputSwitchChangeEvent } from 'primeng/inputswitch';

interface Column {
  field: string;
  header: string;
}

@Component({
  selector: 'app-car-list',
  templateUrl: './car-list.component.html',
  styleUrl: './car-list.component.scss',
  encapsulation: ViewEncapsulation.None
})
export class CarListComponent implements OnInit {

  cols!: Column[]; // table column list
  arrCarList: any = []; // store car's by api
  arrFilterList: any; // store filter to show lable of perticular filter
  selectedColumns!: Column[]; // selected column default list
  arrTableContentTitle = ['category', 'condition', 'status', 'publish_to']; // field to change content to title case

  page = 0;
  limit = 10;

  responsiveOptions: any[] | undefined; // resposive html element of primng

  isExportCSVDialogOpen: boolean = false;
  colsCSVSelected!: any; // table column list
  arrCarListCopy: any = []; // Store CSV data with label

  arrListingStatus = [
    { label: 'live', icon: 'local_shipping' },
    { label: 'sold', icon: 'car_tag', info: 'When a vehicle is sold, you can change its listing status to Sold to remove it from the marketplace and label it as "Sold" for your reporting purposes.\n\nInventory that has been placed and remains in the Sold or Removed status for more than 30 days will be completely removed from these folders and no longer accessible. If you need to keep any vehicle offline for more than 30 days, we suggest that you move these vehicles to the Staging section.' },
    { label: 'removed', icon: 'auto_delete', info: 'if you want to remove a listing from the marketplace, but the vehicle has not been sold, then you can change its listing status to Removed.\n\nInventory that has been placed and remains in the Sold or Removed status for more than 30 days will be completely removed from these folders and no longer accessible. If you need to keep any vehicle offline for more than 30 days, we suggest that you move these vehicles to the Staging folder.' },
    { label: 'staging', icon: 'downloading', info: 'A listing with a status of Staging is not live, but exists for you to keep preparing until you are ready to go live.' }
  ];
  strCurrentCarStatusList = 'live';

  constructor(
    private readonly router: Router,
    readonly commonService: CommonService,
    private readonly csvService: CSVService,
    private readonly httpService: HttpService,
    private readonly messageService: MessageService,
    private readonly confirmationService: ConfirmationService,
    readonly localStorageService: LocalStorageService
  ) { }

  /**
   * Initializes the component by fetching the car list.
   * This method is called automatically by Angular when the component is initialized.
   * It asynchronously retrieves the car list based on the current car status list.
   * 
   * @returns A Promise that resolves to void when the initialization is complete.
   */
  async ngOnInit(): Promise<void> {
    await this.getCarList(this.strCurrentCarStatusList);
  }

  /**
 * Retrieves and processes the car list based on the given status and pagination event.
 * This method handles user authentication, fetches car data, and prepares it for display and export.
 * 
 * @param status - The status of the cars to retrieve (e.g., 'live', 'sold', 'removed', 'staging').
 * @param event - Optional. The table pagination event containing information about the current page and rows per page.
 * @returns A Promise that resolves when the car list has been retrieved and processed.
 *          If there's no car data or the user is redirected, the function may return early.
 */
  async getCarList(status: string, event?: TablePageEvent) {

    // check user type
    const isUserDealer = this.localStorageService.userIsDealer();

    // check user is dealer or store
    let storeOrUser = this.localStorageService.get('store');
    if (!storeOrUser && isUserDealer) {
      this.router.navigateByUrl('/user/store');
      return
    }

    if (!isUserDealer) {
      storeOrUser = this.localStorageService.getObject('user').identifier;
    }

    // check if paginate then update default value
    if (event) {
      this.page = event.first / event.rows;
      this.limit = event.rows;
    }

    // get filter's to set filter field's in column
    this.arrFilterList = (await this.httpService.get('filter') as any).data;

    // get dynamic data from filter
    this.arrCarList = await this.httpService.get(`car/user/${storeOrUser}/${status}?page=${this.page}&limit=${this.limit}`);
    if (!this.arrCarList?.data.length) {
      return
    }

    //copy car list data for csv
    this.arrCarListCopy = JSON.parse(JSON.stringify(this.arrCarList.data));

    // make list for earliest display column 
    const arrTempSequence = ['images', 'category', 'make', 'model', 'year', 'vin_number', 'condition', 'status', 'asking_price'];
    // add in table column list
    this.cols = arrTempSequence.map((key: string) => { return { field: key, header: this.commonService.replaceStringToCapitalize(key) } });
    // check there is other columns in array - yes then include after earlist column list
    Object.keys(this.arrCarList?.data[0]).map((key: string) => {
      const isExist = this.cols.find((element: Column) => element.field === key)
      if (isExist || key === '_id' || key === 'description' || key === 'warrenty_description'
        || key === 'videos' || key === 'images' || key === 'video_url') return // if is exist then return - prevent duplication & remove id as well
      this.cols.push({ field: key, header: this.commonService.replaceStringToCapitalize(key) });
    });
    // add earlist column list in selected column by default
    this.selectedColumns = arrTempSequence.map((key: string) => { return { field: key, header: this.commonService.replaceStringToCapitalize(key) } });

    //export CSV
    // copy cols array to select field on export csv
    this.colsCSVSelected = JSON.parse(JSON.stringify(this.cols));
    this.colsCSVSelected = this.colsCSVSelected.filter((col: any) => !['active', 'location', 'createdBy', 'category'].includes(col.field));
    this.arrCarListCopy.forEach((car: any) => {
      Object.keys(car).forEach((key: string) => {
        if (this.checkFieldIsFromFilter(key)) {
          car[key] = this.checkFilterTitleFromFilterList(key, car[key]);
        }
      })
    });
  }

  /**
   * Updates the active status of a car.
   * 
   * This asynchronous function sends a request to update the active status of a car
   * based on the provided input switch event. If the update is successful, it displays
   * a success message and refreshes the car list.
   * 
   * @param event - The InputSwitchChangeEvent containing the new active status.
   * @param id - The unique identifier of the car to be updated.
   * @returns A Promise that resolves when the car status has been updated and the car list has been refreshed.
   */
  async activeStatusCar(event: InputSwitchChangeEvent, id: string) {
    // send request to active / deactive car data
    const result: any = await this.httpService.patch('car/status/' + id, { active: event.checked });
    if (result.success) {
      this.messageService.add({ severity: 'secondary', summary: 'Success', detail: 'Car status is updated!' });
      await this.getCarList(this.strCurrentCarStatusList);
    }
  }

  /**
   * Finds the primary image of a car based on the lowest position value.
   * 
   * @param car - An array of image objects representing the car's images.
   *              Each image object is expected to have a 'position' property.
   * 
   * @returns The image object with the lowest position value, which is considered
   *          the primary image. If no images are found, it returns undefined.
   */
  findCarPrimaryImage(car: any) {
    return car.find((image: any) => image.position === Math.min(...car.map((img: any) => img.position)));
  }

  /**
   * Checks if the given header should be converted to title case.
   * 
   * This function determines whether the provided header string should be
   * transformed into title case based on a predefined list of headers
   * stored in the `arrTableContentTitle` array.
   *
   * @param header - The header string to check for title case conversion.
   * @returns The matching header string if found in the predefined list, otherwise undefined.
   */
  checkTitleShouldToChange(header: string) {
    return this.arrTableContentTitle.find((search: string) => search === header);
  }

  /**
   * Checks if the given field is a filter field based on the component's filter list.
   * 
   * This function determines whether a field is considered a filter field by searching
   * for it in the component's filter list and checking its type. A field is considered
   * a filter field if it exists in the filter list and its type is either 'textbox-list'
   * or 'checkbox'.
   *
   * @param field - The name of the field to check.
   * @returns A boolean value. Returns true if the field is a filter field (exists in the
   *          filter list and has a type of 'textbox-list' or 'checkbox'), false otherwise.
   */
  checkFieldIsFromFilter(field: string) {
    const filter = this.arrFilterList.find((element: any) => element.name.value === field);
    return filter && (filter.type === 'textbox-list' || filter.type === 'checkbox');
  }

  /**
   * Checks if the given data is an array.
   * 
   * This function uses the Array.isArray() method to determine whether
   * the provided data is an array or not.
   *
   * @param data - The value to be checked. Can be of any type.
   * @returns A boolean value. Returns true if the data is an array, false otherwise.
   */
  checkFieldIsArray(data: any): boolean {
    return Array.isArray(data);
  }

  /**
   * Retrieves the label for a given filter and title from the filter list.
   * 
   * This function searches the component's filter list for a matching filter and title,
   * and returns the corresponding label. If no match is found, it returns a default value.
   *
   * @param filter - The name of the filter to search for in the filter list.
   * @param title - The value to match within the found filter's data.
   * @returns The label corresponding to the matched filter and title. 
   *          Returns '---' if no match is found.
   */
  checkFilterTitleFromFilterList(filter: string, title: string) {
    const data = this.arrFilterList.find((element: any) => element.name.value === filter).data.find((data: any) => data.value === title);
    if (!data) {
      return '---';
    }
    return data.label;
  }

  /**
   * Downloads a CSV file containing selected car data.
   * 
   * This method filters the selected columns from `colsCSVSelected`,
   * extracts the headers and fields, and then uses the `csvService`
   * to generate and download a CSV file named 'cars.csv' with the
   * filtered car data.
   * 
   * @returns {void} This method doesn't return a value but triggers
   *                 a file download in the user's browser.
   */
  downloadCSV(): void {
    const header = this.colsCSVSelected.filter((row: any) => row.selected).map((row: Column) => { return row.header });
    const field = this.colsCSVSelected.filter((row: any) => row.selected).map((row: Column) => { return row.field });
    this.csvService.downloadFile(this.arrCarListCopy, header, field, 'cars.csv');
  }

  /**
   * Generates and downloads a CSV file formatted for AutoTrader with car listing information.
   * 
   * This method processes the car list data, formats it according to AutoTrader's specifications,
   * and initiates a download of the resulting CSV file. It includes details such as car
   * specifications, VIN information, photo URLs, and last modified dates.
   * 
   * The function performs the following steps:
   * 1. Defines the CSV header and corresponding field names.
   * 2. Iterates through the car list, extracting and formatting relevant information.
   * 3. Identifies the primary photo and additional photos for each car.
   * 4. Formats dates according to AutoTrader's requirements.
   * 5. Generates the CSV file and triggers its download.
   * 
   * @returns {void} This method doesn't return a value but triggers a file download
   *                 of 'autotrader.csv' in the user's browser.
   */
  downloadAutoTraderCSV(): void {
    const header = ['Year', 'Make', 'Model', 'Stock Number', 'VIN Number', 'VIN Status', 'Unique Identifier', 'Ad Last Modified Date', 'Photo', 'Photo Last Modified Date', 'Additional Photos', 'Additional Photos Last Modified Date'];
    const field = ['year', 'make', 'model', 'stock_number', 'vin_number', 'vin_status', 'unique_identifier', 'update_at', 'photo', 'photo_date', 'additional_photos', 'additional_photos_date'];
    const arrTemp: any = [];
    this.arrCarListCopy.forEach((car: any) => {
      const minPosition = Math.min(...car.images.map((image: any) => { return image.position }));
      const arrOtherImages = car.images.filter((image: any) => image.position !== minPosition);
      arrTemp.push({
        year: car.year,
        make: car.make,
        model: car.model,
        stock_number: car.stock_number ?? 'n/a',
        vin_number: car.vin_number ?? 'n/a',
        vin_status: car.condition ?? 'n/a',
        unique_identifier: 'stock',
        update_at: dayjs(car.updatedAt).format('MM/DD/YYYY HH:mm:ss'),
        photo: String(car.images[minPosition].url),
        photo_date: dayjs(car.images[minPosition].date).format('MM/DD/YYYY HH:mm:ss'),
        additional_photos: (arrOtherImages.map((image: any) => { return image.url })).join(','),
        additional_photos_date: (arrOtherImages.map((image: any) => { return dayjs(image.date).format('MM/DD/YYYY HH:mm:ss') })).join(','),
      })
    });
    this.csvService.downloadFile(arrTemp, header, field, 'autotrader.csv');
  }

  /**
   * Redirects the user to the car management page after checking their profile status.
   * If the user is not a dealer, it checks if they have an address in their profile.
   * If no address is found, it prompts the user to add an address before proceeding.
   * 
   * @param event - The event object that triggered this function. Used to target the confirmation dialog.
   * @returns A Promise that resolves when the redirection is complete or the confirmation dialog is shown.
   *          The function doesn't return any value explicitly, but may redirect the user to different pages.
   */
  async redirectToAddCar(event: any): Promise<void> {
    if (!this.localStorageService.userIsDealer()) {
      // get user profile
      const user: any = await this.httpService.get('user/profile');
      if (!user.address) {
        this.confirmationService.confirm({
          target: event.target as EventTarget,
          message: 'There is no address in your profile, Kindly add address first to add advertisement?',
          header: 'Caution',
          icon: 'pi pi-exclamation-triangle',
          key: 'addressUser',
          rejectButtonStyleClass: "p-button-text",
          accept: () => {
            this.router.navigateByUrl('/user/profile');
          },
          reject: () => {
          }
        });
        return
      }
    }
    this.router.navigateByUrl('/user/car-manage');
  }
}
