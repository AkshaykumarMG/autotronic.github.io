import { Component, Inject, PLATFORM_ID, inject } from '@angular/core';
import { isPlatformBrowser } from '@angular/common';
import { Router, RouterModule } from '@angular/router';

// firebase token
import { Messaging, deleteToken } from '@angular/fire/messaging';

// custom services
import { AuthService } from '../../utilities/services/auth.service';
import { EventService } from '../../utilities/services/event.service';
import { LocalStorageService } from '../../utilities/services/local-storage.service';

// primeng
import { MenuItem } from 'primeng/api';
import { MenuModule } from 'primeng/menu';
import { DividerModule } from 'primeng/divider';
import { ButtonModule } from 'primeng/button';
import { StyleClassModule } from 'primeng/styleclass';

// angular cdk
import { BreakpointObserver } from '@angular/cdk/layout';
import { HttpService } from '../../utilities/services/http.service';

@Component({
  selector: 'app-index',
  standalone: true,
  imports: [RouterModule, MenuModule, DividerModule, ButtonModule, StyleClassModule],
  templateUrl: './index.component.html',
  styleUrl: './index.component.scss'
})
export class IndexComponent {

  // menu items
  menuLanguage: MenuItem[] | undefined;
  menuMainNavigation: MenuItem[] | undefined;
  menuMainNavigationResponsive: MenuItem[] = [];
  menuLoginUser: any;

  // current login user session
  currentUser: boolean = false;

  // firebase notification
  private messaging = inject(Messaging);

  // user display
  isMobileScreen = this.breakpointObserver.isMatched('(max-width: 599px)');

  constructor(
    private readonly router: Router,
    private readonly httpService: HttpService,
    private readonly authService: AuthService,
    private readonly eventService: EventService,
    @Inject(PLATFORM_ID) private platformId: Object,
    private readonly breakpointObserver: BreakpointObserver,
    private readonly localStorageService: LocalStorageService
  ) { }

  /**
   * Initializes the component by performing several setup tasks.
   * 
   * This method is called automatically by Angular when the component is initialized.
   * It performs the following tasks:
   * 1. Requests the user's location.
   * 2. Initializes component values.
   * 3. Checks if the user is logged in by making an API call.
   * 4. Sets up a subscription to handle navigation user option events.
   * 
   * @returns {Promise<void>} A promise that resolves when all initialization tasks are complete.
   */
  async ngOnInit(): Promise<void> {
    this.requestForUserLocation();
    this.onInitValues();
  
    // check are valid with login
    this.httpService.get('user/identity').then(async (user: any) => {
      if (user) {
        this.currentUser = true;
      }
    }).catch((error: any) => {
      this.currentUser = false;
    });
  
    this.eventService.navUserOptionEvent
      .subscribe(() => {
        this.currentUser = true;
      });
  }

  /**
   * Initializes the navigation menu items for both large screens and responsive layouts.
   * This method sets up the main navigation menu, user authentication options, and creates
   * a responsive version of the main navigation for smaller devices.
   *
   * @returns {void}
   */
  onInitValues(): void {
    // setup option for main navigation for large screen menu
    this.menuMainNavigation = [
      {
        label: 'Home',
        icon: 'pi pi-home',
        routerLink: ''
      },
      {
        label: 'Buy Online',
        icon: 'pi pi-shopping-bag',
        routerLink: 'search'
      },
      {
        label: 'Prequalify Loan',
        icon: 'pi pi-money-bill',
        routerLink: 'prequalify-loan'
      }
    ];

    // setup authentication options
    this.menuLoginUser = [
      {
        label: 'Home',
        type: 'link',
        routerLink: '/user/home',
      },
      {
        label: 'My vehicles',
        type: 'link',
        routerLink: '/user/car-list',
      },
      {
        label: 'Saved Search',
        type: 'link',
        routerLink: '/user/search-alert',
      },
      {
        label: 'Saved Vehicle',
        type: 'link',
        routerLink: '/user/favourite-vehicle',
      },
      {
        label: 'Price Alert',
        type: 'link',
        routerLink: '/user/price-alert',
      },
      {
        label: 'Account Settings',
        type: 'link',
        routerLink: '/user/profile',
      },
      {
        label: 'Logout',
        type: 'function'
      }];

    // setup dynamic responsive menu for small devices
    this.menuMainNavigationResponsive = [];
    this.menuMainNavigation.forEach((item: MenuItem) => {
      this.menuMainNavigationResponsive.push({
        label: item.label,
        icon: item.icon,
        routerLink: item.routerLink
      })
    });
    this.menuMainNavigationResponsive = [...this.menuMainNavigationResponsive, {
      label: 'Authentication',
      auth: true
    }];
  }


  /**
   * Performs the user logout process.
   * 
   * This asynchronous function handles the complete logout procedure, which includes:
   * 1. Deleting the user's record from the authentication database.
   * 2. Removing the Firebase Cloud Messaging (FCM) token.
   * 3. Redirecting the user to the login page.
   * 4. Reloading the page to ensure all user-specific data is cleared.
   * 
   * @async
   * @returns {Promise<void>} A promise that resolves when the logout process is complete.
   * @throws Will throw an error if any step of the logout process fails.
   */
  async onLogoutUser(): Promise<void> {
    // Delete record from auth db
    await this.authService.OnLogout();
  
    // Remove FCM token
    await this.deleteFCMToken();
  
    // Redirect to login
    this.router.navigateByUrl('/login');
  
    // reload page
    location.reload();
  }

  /**
   * Requests and stores the user's current geolocation.
   * 
   * This function checks if the browser supports geolocation and if the user's location
   * is not already stored. If these conditions are met, it requests the user's current
   * position, stores it in local storage, and refreshes the page.
   * 
   * @returns {void}
   * 
   * @remarks
   * This function only executes in browser environments and relies on the availability
   * of the Geolocation API. It uses the LocalStorageService to store the location data.
   * 
   * @throws {GeolocationPositionError} If there's an error obtaining the geolocation.
   */
  requestForUserLocation(): void {
    if (isPlatformBrowser(this.platformId)) {
      const userLocation = this.localStorageService.get('location');
      if (!userLocation) {
        if (navigator.geolocation) {
          navigator.geolocation.getCurrentPosition((position: GeolocationPosition) => {
            if (position) {
              const latitude: number = position.coords.latitude;
              const longitude: number = position.coords.longitude;
              const newLocation = {
                lat: latitude,
                lng: longitude
              };
              this.localStorageService.setObject('location', newLocation)
              location.reload(); // refresh forcefully
            }
          },
            (error: GeolocationPositionError) => console.error(error));
        } else {
          console.log('Geolocation is not supported by this browser.');
        }
      }
    }
  }


  /**
   * Deletes the Firebase Cloud Messaging (FCM) token associated with the current device.
   * 
   * This function removes the FCM token, which is used for push notifications.
   * It's important to note that if you're storing FCM tokens in your Firestore database,
   * you should also update the database to reflect this deletion.
   * 
   * @async
   * @returns {Promise<void>} A promise that resolves when the token has been successfully deleted.
   * @throws Will throw an error if the token deletion fails.
   */
  async deleteFCMToken(): Promise<void> {
    // We can also delete fcm tokens, make sure to also update this on your firestore db if you are storing them as well
    await deleteToken(this.messaging);
  }
}
