  /**
   * Initializes the component by setting up initial values and handling query parameters.
   * 
   * This method performs the following tasks:
   * 1. Calls onInitValues to set up initial component data.
   * 2. Subscribes to route query parameters to pre-fill car details if available.
   * 3. Enables the model field in the form if query parameters are present.
   * 4. Filters and sorts the available car models based on the selected make.
   * 
   * @returns {Promise<void>} A promise that resolves when the initialization is complete.
   */
  async ngOnInit(): Promise<void> {
    // initiate page operations
    await this.onInitValues();

    // check car detail avalable by query string
    this.route.queryParams.subscribe((params: any) => {
      if (Object.keys(params).length > 0) {
        // fill car detail form
        this.formCarDetail.patchValue({
          make: params.make,
          model: params.model,
          year: Number(params.year),
          price: Number(params.price)
        });

        // enable model field
        this.formCarDetail.get('model')?.enable();

        // find related models to make and set
        const dependantData = this.arrModelCopy.filter((element: any) => element.join === params.make);
        this.arrModel = dependantData.sort((a: any, b: any) => a.value > b.value ? 1 : -1);
      }
    });
  }

  /**
   * Initializes the component by fetching and setting up various data needed for the application.
   * This includes car make and model data, year options, and location information.
   * 
   * @returns A Promise that resolves when all initialization tasks are complete.
   * @throws Will throw an error if any of the HTTP requests or data processing fails.
   */
  async onInitValues(): Promise<void> {
    // get filters for make & model
    const filters = (await this.httpService.get('filter') as any).data;

    // Extract make data and sort it alphabetically
    this.arrMake = filters.find((ele: any) => ele.name.value === 'make').data.sort((a: any, b: any) => a.value > b.value ? 1 : -1);

    // Extract model data
    this.arrModelCopy = filters.find((ele: any) => ele.name.value === 'model').data;

    // Generate a list of years from the current year to 50 years ago
    const now = new Date().getUTCFullYear();
    this.arrYear = Array(now - (now - 50)).fill('').map((v, idx) => now - idx);

    // Disable the model field by default
    this.formCarDetail.get('model')?.disable();

    // Fetch city and province data from a static CSV file
    const csv: any = await this.httpService.getWithOutsideURLCSV('assets/csv/canadacities.csv');

    // Convert CSV data to JSON and extract province IDs
    const province = this.commonService.CSVToJSON(csv).map((cities: any) => { return cities.province_id });

    // Create a list of unique provinces and format them for the dropdown
    this.data_province = [...new Set(province)].map((cities: any) => { return { label: cities, value: cities.toLowerCase() } });
  }


  /**
 * Handles the change event of the car make dropdown.
 * It filters the related models based on the selected make and sorts them alphabetically.
 * It also enables the model field in the car details form.
 *
 * @param event - The DropdownChangeEvent object containing the selected make value.
 */
  onChangeMake(event: DropdownChangeEvent): void {

    // find related models to make and set
    const dependantData = this.arrModelCopy.filter((element: any) => element.join === event.value);
    this.arrModel = dependantData.sort((a: any, b: any) => a.value > b.value ? 1 : -1);

    // enable model field
    this.formCarDetail.get('model')?.enable();
  }

  /**

 */
  async onCheckCreditReport(): Promise<void> {

    // check result is already displayed, if not then show result first otherwise update
    if (this.isCarDetailStepCompleted) {
      // end button loading
      this.isCreditReportLoading = true;

      //check income duration, if false then remove previous employer details
      const dataFinanceForm = this.formFinanceDetail.value;

      // check residency status, if other then not approved application
      if (dataFinanceForm.residency_status === 'other') {

        this.arrResultCreditReport = [
          {
            lender: 'AutoTronic',
            score: 0,
            status: 'Not Approved',
            term: 0,
            amount: 0,
            apr: 0,
            monthly_payment: 0,
            notes: ['Unfortunately no financing option is avaialble for non-canadians']
          }
        ];

        //   score: 0,
        //   status: 'Not Approved',
        //   notes: ['Unfortunately no financing option is avaialble for non-canadians']
        // })

        // refresh chart
        this.setupScoreChart();

        // end button loading
        this.isCreditReportLoading = false;

        return
      }

      try {
        // update car search data for report add in database
        const responseResult: any = await this.httpService.patch(`vehicle-finance-application/${this.strUpdateVechicleApplicationID}`,
          { ...dataFinanceForm, user: this.localStorageService.getObject('user').identifier });

        // if we add data then display result
        if (responseResult.success) {

          // load credit report result
          this.arrResultCreditReport = responseResult.data.result;

          // change down payment
          this.formCarDetail.patchValue({
            down_payment: dataFinanceForm.down_payment
          });

          // load random credit report result
          // this.formCreditReportResult.patchValue({
          //   score: responseResult.data.result.score,
          //   status: responseResult.data.result.status,
          //   term: responseResult.data.result.term,
          //   amount: responseResult.data.result.amount,

        this.messageService.add({ severity: 'error', summary: 'Error', detail: error });

        this.isCreditReportLoading = false;

      this.httpService.get('user/profile').then(async (user: any) => {
        if (user) {
          // split first and last name from user name
          const first_name = user.username.split(' ').slice(0, -1).join(' ');
          const last_name = user.username.split(' ').slice(-1).join(' ');

          // fill user information form
          this.formContactDetail.patchValue({
            email: user.email,
            phone: user.phone,
            first_name,
            last_name,
          });

          const dataCar = this.formCarDetail.value;
          const dataNecessarFields = this.formNecesseryFields.value;
          try {

            // add car search data for report add in database
            const responseResult: any = await this.httpService.post('vehicle-finance-application', {
              ...dataCar,
              ...dataNecessarFields,
              category: 'car',
              user: user._id
            });

              down_payment: dataCar.down_payment,
              estimated_credit_score: responseResult.data.estimated_credit_score,
              monthly_payment: responseResult.data.monthly_payment,
              annual_income: responseResult.data.annual_income,
            });

              this.strUpdateVechicleApplicationID = responseResult.data._id;

              this.isCarDetailStepCompleted = true;

              this.isCreditReportLoading = false;

            if (error === 'Missing Information') {
              this.isNecesseryFieldDialogOpen = true;
              return
            }
            this.messageService.add({ severity: 'error', summary: 'Error', detail: error });
          } finally {
            this.isCreditReportLoading = false;
          }
        }
      }).catch((error: HttpErrorResponse) => {
        if (error.status === 401) {
          // open credit infor dialog
          this.isCreditInfoDialogOpen = true;

          // end button loading
          this.isCreditReportLoading = false;
        }
      });
    }

          down_payment: result.result.down_payment,
          annual_income: result.result.annual_income,
          estimated_credit_score: result.result.estimated_credit_score,

  /**
 * Handles the display of an assistance popup to the user.
 *
 * This function is called when the user requests assistance.
 * It sets the `isCreditReportLoading` flag to `true` to indicate that the button is loading.
 * Then, it uses the `ConfirmationService` to display a confirmation popup with a custom message.
 * After the user acknowledges the message, the `isCreditReportLoading` flag is set to `false` to indicate that the button is no longer loading.
 *
 * @remarks
 * This function is used to provide a way for the user to request assistance from the application.
 *
 * @returns {void}
 */
  async onShowAssistPopup(): Promise<void> {
    // start button loading
    this.isHelpFinanceButtonLoading = true;

    try {
      // if user already asked for help then add sub entry in vehicle finance application
      if (this.isHelpFinanceApplicationSubscribed) {
        // add sub entry in vehicle finance application
        await this.httpService.patch(`vehicle-finance-application/sub/${this.strUpdateVechicleApplicationID}`, {
          ...this.formFinanceDetail.value,
          result: this.arrResultCreditReport // this.formCreditReportResult.value
        });
      } else {
        // call api to send help request
        await this.httpService.post('finance-help', {
          vehicle_finance_application: this.strUpdateVechicleApplicationID
        });

        // set help finance application subscribed to add sub entry in vehicle finance application
        this.isHelpFinanceApplicationSubscribed = true;
      }

      this.isHelpFinanceButtonDisabled = true;

      this.confirmationService.confirm({

        message: 'One of our agents will contact you shortly.',
      });

      });
    } finally {
      // end button loading
      this.isHelpFinanceButtonLoading = false;
    }
  }
