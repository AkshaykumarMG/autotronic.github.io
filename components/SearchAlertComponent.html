Finds and formats the search label and value for a given filter.

This function processes a filter object to determine its corresponding label and value. It handles special cases for 'Distance' filters and processes 'checkbox' and 'textbox-list' type filters differently.

import { Component } from '@angular/core';
import { Router } from '@angular/router';

// custom service
import { HttpService } from '../../../utilities/services/http.service';

// primeng
import { MessageService } from 'primeng/api';

@Component({
  selector: 'app-search-alert',
  templateUrl: './search-alert.component.html',
  styleUrl: './search-alert.component.scss'
})
export class SearchAlertComponent {

  // property to store the alert subscription data fetched from the API
  arrAlertSubscribe: Array<any> = [];
  // property to store the filters fetched from the API
  arrFilters: any;

  constructor(
    private readonly router: Router,
    private readonly httpService: HttpService,
    private readonly messageService: MessageService
  ) { }

  /**
   * Initializes the component by fetching filters and alert subscriptions.
   * This method is called automatically by Angular when the component is initialized.
   * 
   * @returns A Promise that resolves when the initialization is complete.
   */
  async ngOnInit(): Promise<void> {
      // Fetch filters from the API and store them in arrFilters
      this.arrFilters = (await this.httpService.get('filter') as any).data;
  
      // Fetch alert subscriptions for the user
      await this.getAlertSubscribe();
  }

  /**
   * Fetches alert subscriptions for the current user from the server.
   * 
   * This asynchronous method sends a GET request to the 'alert-subscribe/user' endpoint
   * using the httpService. Upon successful response, it updates the arrAlertSubscribe
   * property with the received data.
   * 
   * @returns {Promise<void>} A promise that resolves when the alert subscriptions have been fetched and stored.
   */
  async getAlertSubscribe(): Promise<void> {
    await this.httpService.get('alert-subscribe/user').then((response: any) => {
      this.arrAlertSubscribe = response;
    });
  }

  /**
   * Navigates to the detail page for a specific alert subscription.
   * 
   * This method is triggered when a user clicks on an alert subscription item.
   * It uses the Angular Router to navigate to the detail page, passing the
   * subscription's ID as a route parameter.
   *
   * @param {string} id - The unique identifier of the alert subscription to view.
   * @returns {void} This method does not return a value.
   */
  onClickAlertSubscribe(id: string): void {
    this.router.navigate(['/detail/', id]);
  }

  /**
   * Removes an alert subscription for a specific vehicle.
   * 
   * This asynchronous method is triggered when a user attempts to unsubscribe from a vehicle alert.
   * It prevents the default event behavior, sends a delete request to the server, and updates the UI accordingly.
   *
   * @param {any} event - The DOM event object triggered by the user action.
   * @param {string} id - The unique identifier of the alert subscription to be removed.
   * @returns {Promise<void>} A promise that resolves when the unsubscribe process is complete.
   */
  async onRemoveAlertSubscribe(event: any, id: string): Promise<void> {
    event.stopPropagation();
    event.preventDefault();
    await this.httpService.delete(`alert-subscribe/${id}`)
      .then((response: any) => {
        if (response.success) {
          this.messageService.add({ severity: 'secondary', summary: 'Success', detail: 'Vehicle Unsubscribed successfully!' });
          this.getAlertSubscribe();
        }
      });
  }

  /**
   * Finds and formats the search label and value for a given filter.
   * 
   * This function processes a filter object to determine its corresponding label and value.
   * It handles special cases for 'Distance' filters and processes 'checkbox' and 'textbox-list' type filters differently.
   * 
   * @param {any} filter - The filter object to process. It should contain at least a 'name' property and a 'value' property.
   * @returns {{ label: string, value: string }} An object containing:
   *   - label: The display label for the filter.
   *   - value: The formatted value of the filter. For 'Distance', it's either 'National' or a distance in km.
   *            For other types, it's a string representation of the selected option(s).
   */
  findSearchLabelByValue(filter: any): { label: string; value: string; } {
    if (filter.name === 'Distance') {
      return { label: 'Distance', value: filter.value === 'national' ? 'National' : `${filter.value} km` };
    }
    const findFilter = this.arrFilters.find((flr: any) => flr.name.value === filter.name);
    let value = '';
    if (findFilter && findFilter.type === 'checkbox') {
      findFilter.data.forEach((data: any) => {
        if (filter.value.includes(data.value)) {
          value += value ? `, ${data.label}` : `${data.label}`;
        }
      })
    }
    if(findFilter && findFilter.type === 'textbox-list') {
      findFilter.data.forEach((data: any) => {
        if (filter.value === data.value) {
          value = data.label ;
        }
      });
    }
    return { label: findFilter.name.label, value };
  }

}
