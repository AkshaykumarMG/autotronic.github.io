import { Component } from '@angular/core';
import { Router } from '@angular/router';

// custom service
import { HttpService } from '../../../utilities/services/http.service';

// primeng
import { MessageService } from 'primeng/api';

@Component({
  selector: 'app-favourite-vehicle',
  templateUrl: './favourite-vehicle.component.html',
  styleUrl: './favourite-vehicle.component.scss'
})
export class FavouriteVehicleComponent {

  // array to store the user's favorite vehicles
  arrFavouriteVehicle: Array<any> = [];

  constructor(
    private readonly router: Router,
    private readonly httpService: HttpService,
    private readonly messageService: MessageService
  ) { }

  /**
   * Lifecycle hook that is called after data-bound properties of a directive are initialized.
   * 
   * This method is part of Angular's component lifecycle. It initializes the component by
   * fetching the user's favorite vehicles when the component is created.
   *
   * @returns {Promise<void>} A promise that resolves when the initialization process is complete,
   *                          specifically when the favorite vehicles have been retrieved.
   */
  async ngOnInit(): Promise<void> {
    await this.getFavouriteVahicle();
  }

  /**
   * Retrieves the list of favorite vehicles for the current user.
   * 
   * This asynchronous function makes a GET request to the 'user/favourite-list' endpoint
   * using the HttpService. Upon successful retrieval, it updates the component's
   * arrFavouriteVehicle array with the response data.
   *
   * @returns {Promise<void>} A promise that resolves when the favorite vehicles have been fetched and stored.
   */
  async getFavouriteVahicle(): Promise<void> {
    await this.httpService.get('user/favourite-list').then((response: any) => {
      this.arrFavouriteVehicle = response;
    });
  }

  /**
   * Navigates to the detail page of a favorite vehicle.
   * 
   * This function is triggered when a user clicks on a favorite vehicle in the list.
   * It uses Angular's Router to navigate to the detail page of the selected vehicle.
   *
   * @param id - The unique identifier of the favorite vehicle to be viewed in detail.
   * @returns void This function doesn't return a value.
   */
  onClickFavouriteVehicle(id: string): void {
    this.router.navigate(['/detail/', id]);
  }

  /**
   * Removes a vehicle from the user's favorites list.
   * 
   * This asynchronous function sends a PATCH request to the server to remove the specified vehicle
   * from the user's favorites. If successful, it displays a success message and refreshes the
   * list of favorite vehicles.
   *
   * @param event - The event object from the click event, used to prevent event propagation.
   * @param id - The unique identifier of the vehicle to be removed from favorites.
   * @returns A Promise that resolves when the removal process is complete.
   */
  async onRemoveFavourite(event: any, id: string): Promise<void> {
    event.stopPropagation();
    // Make a patch request to the server to update the user's favourites list
    await this.httpService.patch('user/favourite', { id, mode: 'remove' })
      .then((response: any) => {
        // If the request is successful, refresh current favourite array list
        if (response.success) {
          this.messageService.add({ severity: 'secondary', summary: 'Success', detail: 'Vehicle Unsubscribed successfully!' });
          this.getFavouriteVahicle();
        }
      });
  }
}
