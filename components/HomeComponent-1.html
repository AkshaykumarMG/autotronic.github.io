import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';

// custom service
import { HttpService } from '../../../utilities/services/http.service';
import { LocalStorageService } from '../../../utilities/services/local-storage.service';

@Component({
  selector: 'app-home',
  templateUrl: './home.component.html',
  styleUrl: './home.component.scss'
})
export class HomeComponent implements OnInit {

  // string count total car of store or user
  strCountCar: number | undefined;

  constructor(
    private readonly router: Router,
    private readonly httpService: HttpService,
    private readonly localStorageService: LocalStorageService
  ) { }
  
  /**
 * Initializes the component by fetching and setting the total car count.
 * 
 * This method checks if the user is a dealer or not. If the user is a dealer,
 * it retrieves the total car count for their store. If not, it retrieves the
 * total car count for the user.
 * 
 * @returns {Promise<void>} A promise that resolves when the initialization is complete.
 */
  async ngOnInit(): Promise<void> {

    // check if user is dealer, if yes then get total car count from store else from user
    const isUserDealer = this.localStorageService.userIsDealer();
    if (isUserDealer) {
      // get store from local storage
      let store = this.localStorageService.get('store');
      if (!store) {
        const stores: any = (await this.httpService.get('user/store') as any).store;
        store = stores[0]._id;
      }
      // get total car count from store
      this.strCountCar = await this.httpService.get(`car/count/store/${store}`) as number;
    } else {
      const user = this.localStorageService.getObject('user').identifier;
      this.strCountCar = await this.httpService.get(`car/count/user/${user}`) as number;
    }
  }
}
