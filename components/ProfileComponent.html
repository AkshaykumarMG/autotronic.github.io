import { Component, OnInit } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { environment } from '../../../../environments/environment';

// custom service
import { HttpService } from '../../../utilities/services/http.service';
import { LocalStorageService } from '../../../utilities/services/local-storage.service';
import { CSVService } from '../../../utilities/services/csv.service';

// primeng
import { MessageService } from 'primeng/api';

@Component({
  selector: 'app-profile',
  templateUrl: './profile.component.html',
  styleUrl: './profile.component.scss'
})
export class ProfileComponent implements OnInit {

  // form user information
  formUserInfo = new FormGroup({
    email: new FormControl(null, [Validators.required, Validators.email]),
    phone: new FormControl(null, [Validators.required]),
    username: new FormControl(null, [Validators.required])
  });

  // form change password
  formPassword = new FormGroup({
    old_password: new FormControl(null, [Validators.required, Validators.minLength(10)]),
    new_password: new FormControl(null, [Validators.required, Validators.minLength(10)]),
    compare_password: new FormControl(null, [Validators.required, Validators.minLength(10)]),
  });

  // user address form
  formAddress = new FormGroup({
    unit: new FormControl(null, [Validators.required]),
    street: new FormControl(null, [Validators.required]),
    city: new FormControl(null, [Validators.required]),
    state: new FormControl(null, [Validators.required]),
    country: new FormControl('Canada', [Validators.required]),
    postal_code: new FormControl(null, [Validators.required])
  });
  arrCity: any;
  arrState = [
    { label: 'Alberta', value: 'AB' },
    { label: 'British Columbia', value: 'BC' },
    { label: 'New Brunswick', value: 'NB' },
    { label: 'Manitoba', value: 'MB' },
    { label: 'Newfoundland and Labrador', value: 'NL' },
    { label: 'Northwest Territories', value: 'NT' },
    { label: 'Nova Scotia', value: 'NS' },
    { label: 'Nunavut', value: 'NU' },
    { label: 'Ontario', value: 'ON' },
    { label: 'Quebec', value: 'QC' },
    { label: 'Prince Edward Island', value: 'PE' },
    { label: 'Saskatchewan', value: 'SK' },
    { label: 'Yukon', value: 'YT' },
  ]; // state list for address
  
  constructor(
    private readonly csvService: CSVService,
    private readonly httpService: HttpService,
    private readonly messageService: MessageService,
    private readonly localStorageService: LocalStorageService,
  ) { }

  /**
   * Initializes the component by fetching user data and populating city information.
   * 
   * This method is called automatically when the component is initialized.
   * It performs two main tasks:
   * 1. Retrieves the current user's information.
   * 2. Loads and processes city data from a CSV file.
   * 
   * @returns {Promise<void>} A promise that resolves when the initialization is complete.
   */
  async ngOnInit(): Promise<void> {
    // get current user
    await this.getUser();
  
    // find city as per postal code from static CSV file
    const csv: any = await this.httpService.getWithOutsideURLCSV('assets/csv/canadacities.csv');
    this.arrCity = this.csvService.CSVToJSON(csv).map((cities: any) => { return { city: cities.city_ascii } });
  }

  /**
   * Retrieves the current user's profile information and populates the user information and address forms.
   * 
   * This asynchronous method fetches the user profile data from the server and updates
   * the corresponding form fields with the retrieved information. It populates both
   * the user information form and, if available, the address form.
   * 
   * @async
   * @returns {Promise<void>} A promise that resolves when the user data has been fetched and the forms have been updated.
   * @throws Will throw an error if the HTTP request fails or if there's an issue updating the forms.
   */
  async getUser(): Promise<void> {
    // get user profile
    const user: any = await this.httpService.get('user/profile');
  
    // fill user information form
    this.formUserInfo.patchValue({
      email: user.email,
      phone: user.phone,
      username: user.username
    });
  
    if(user.address) {
      // fill address form 
      this.formAddress.patchValue({
        unit: user.address.unit,
        street: user.address.street,
        city: user.address.city,
        state: user.address.state,
        country: user.address.country,
        postal_code: user.address.postal_code
      });
    }
  }

  /**
   * Submits a request to change the user's password.
   * 
   * This method retrieves the password data from the form, sends a POST request
   * to change the password, and handles the response by displaying appropriate messages.
   * 
   * @async
   * @returns {Promise<void>} A promise that resolves when the password change process is complete.
   * @throws Will display an error message if the password change request fails.
   */
  async submitChangePassword(): Promise<void> {
    const data = this.formPassword.value;
    await this.httpService.post('user/change-password', data).then((response: any) => {
      if (response.success) {
        this.messageService.add({ severity: 'secondary', summary: 'Done', detail: 'Password Changed Successfully' });
      }
    }).catch((error: any) => {
      this.messageService.add({ severity: 'error', summary: 'Fail', detail: error as any });
    });
  }


  /**
   * Submits and validates the user's address using Google Maps API, then updates the user's address in the system.
   * 
   * This method performs the following steps:
   * 1. Retrieves the address data from the form.
   * 2. Validates the address using Google Maps API.
   * 3. If the address is valid, it extracts the latitude and longitude.
   * 4. Updates the user's address in the system with the validated information.
   * 
   * @async
   * @returns {Promise<void>} A promise that resolves when the address submission process is complete.
   * @throws Will display an error message if the address is invalid or if the update request fails.
   */
  async submitAddress(): Promise<void> {
    // get address from form
    const data = this.formAddress.value;

    // verify address by google map
    const url = `https://addressvalidation.googleapis.com/v1:validateAddress?key=${environment.google_key}`;
    const requestBody = {
      address: {
        "regionCode": "CA",
        "addressLines": [`${data?.unit!} ${data?.street!}`, `${data?.city} ${data?.state} ${data?.postal_code}`]
      }
    }

    // call google location api to get location lat lng
    const response: any = await this.httpService.postWithOutsideURL(url, requestBody);

    // if address is correct then show map with maker
    let location: any;
    if (response.result.address) {
      location = { lat: response.result.geocode.location.latitude, lng: response.result.geocode.location.longitude };
    } else {
      this.messageService.add({ severity: 'error', summary: 'Fail', detail: 'Address is not correct, kindly check all fields.' });
      return
    }

    const userID = this.localStorageService.getObject('user').identifier;
    await this.httpService.patch(`user/address/${userID}`, { ...data, lat: location.lat, lng: location.lng }).then((response: any) => {
      if (response.success) {
        this.messageService.add({ severity: 'secondary', summary: 'Done', detail: 'Address Submit Successfully' });
      }
    }).catch((error: any) => {
      this.messageService.add({ severity: 'error', summary: 'Fail', detail: error });
    });
  }

}
