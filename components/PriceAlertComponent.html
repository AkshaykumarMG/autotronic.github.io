import { Component } from '@angular/core';
import { Router } from '@angular/router';

// custom service
import { HttpService } from '../../../utilities/services/http.service';

// primeng
import { MessageService } from 'primeng/api';

@Component({
  selector: 'app-price-alert',
  templateUrl: './price-alert.component.html',
  styleUrl: './price-alert.component.scss'
})
export class PriceAlertComponent {

  // property to store the price alert data fetched from the API
  arrPriceAlert: Array<any> = [];

  constructor(
    private readonly router: Router,
    private readonly httpService: HttpService,
    private readonly messageService: MessageService
  ) { }

  /**
   * Lifecycle hook that is called after data-bound properties of a directive are initialized.
   * 
   * This method is part of Angular's component lifecycle. It initializes the component by
   * fetching price alerts when the component is created.
   *
   * @returns {Promise<void>} A promise that resolves when the initialization process,
   * including fetching price alerts, is complete. The function doesn't return any value directly.
   */
  async ngOnInit(): Promise<void> {
    await this.getPriceAlerts();
  }

  /**
   * Fetches price alerts for the current user.
   * 
   * This asynchronous function makes an HTTP GET request to retrieve price alerts
   * associated with the current user. The retrieved alerts are then stored in the
   * component's `arrPriceAlert` array.
   * 
   * @returns {Promise<void>} A promise that resolves when the price alerts have been
   * fetched and stored in the component's state. The function doesn't return any value directly.
   * 
   * @throws {Error} May throw an error if the HTTP request fails or if there's an issue
   * processing the response.
   */
  async getPriceAlerts(): Promise<void> {
    await this.httpService.get('alert-price/user').then((response: any) => {
      this.arrPriceAlert = response;
    });
  }

  /**
   * Removes a price alert for a specific vehicle and updates the UI.
   * 
   * This asynchronous function handles the removal of a price alert when triggered by a user action.
   * It prevents event propagation, sends a delete request to the server, and updates the UI accordingly.
   *
   * @param event - The event object from the user interaction, used to stop event propagation.
   * @param id - The unique identifier of the price alert to be removed.
   * @returns A Promise that resolves when the alert removal process is complete.
   */
  async onRemoveAlert(event: any, id: string): Promise<void> {
    event.stopPropagation();
    await this.httpService.delete(`alert-price/${id}`).then((response: any) => {
      if(response.success) {
        this.messageService.add({ severity: 'secondary', summary: 'Success', detail: 'Vehicle Unsubscribed successfully!' });
        this.getPriceAlerts();
      }
    });
  }

  /**
   * Navigates to the detail page of a specific alert when clicked.
   * 
   * @param id - The unique identifier of the alert to be viewed in detail.
   * @returns void This function doesn't return a value.
   */
  onClickAlert(id: string): void {
    this.router.navigate(['/detail/', id]);
  }

}
