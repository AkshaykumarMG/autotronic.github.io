import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { CommonModule } from '@angular/common';
import { FormControl, FormGroup, ReactiveFormsModule, Validators } from '@angular/forms';

// custom service
import { HttpService } from '../../utilities/services/http.service';

// map
import { GoogleMap, MapMarker } from '@angular/google-maps';

// primeng
import { DropdownModule } from 'primeng/dropdown';
import { InputTextModule } from 'primeng/inputtext';
import { ButtonModule } from 'primeng/button';
import { PaginatorModule, PaginatorState } from 'primeng/paginator';
import { DividerModule } from 'primeng/divider';
import { ChipModule } from 'primeng/chip';
import { AvatarModule } from 'primeng/avatar';

@Component({
  selector: 'app-dealer-detail',
  standalone: true,
  imports: [CommonModule, ReactiveFormsModule, GoogleMap, MapMarker, DropdownModule, InputTextModule, ButtonModule, PaginatorModule, DividerModule, ChipModule, AvatarModule],
  templateUrl: './dealer-detail.component.html',
  styleUrl: './dealer-detail.component.scss'
})
export class DealerDetailComponent implements OnInit {

  // defined variables
  objStoreInfo: any;
  storeID!: string;
  arrMake: any;
  arrModel: any;
  arrModelCopy: any;
  arrCars: any;

  // pagination
  page: number = 0;
  limit: number = 15;

  // filter form
  formFilter: FormGroup = new FormGroup({
    make: new FormControl<string | null>(null, [Validators.required]),
    model: new FormControl<string | null>(null, [Validators.required]),
    max_price: new FormControl<string | null>(null)
  })

  // map
  center: google.maps.LatLngLiteral = { lat: 24, lng: 12 };
  zoom = 14;
  markerOptions: google.maps.MarkerOptions = { draggable: false };
  markerPositions: google.maps.LatLngLiteral[] = [];

  constructor(
    private readonly router: Router,
    private readonly route: ActivatedRoute,
    private readonly httpService: HttpService
  ) {
    this.route.params.subscribe(async params => {
      this.storeID = params['id'];
    });
  }

  /**
   * Initializes the component by fetching store information, setting up map data,
   * retrieving filter options for make and model, and fetching initial vehicle data.
   * 
   * This method is called automatically by Angular when the component is initialized.
   * It performs the following tasks:
   * 1. Fetches store information based on the store ID.
   * 2. Sets up the map center and marker position using the store's coordinates.
   * 3. Retrieves and sets up filter options for make and model.
   * 4. Disables the model field in the filter form by default.
   * 5. Fetches the initial list of vehicles for the store.
   * 
   * @returns {Promise<void>} A promise that resolves when all initialization tasks are complete.
   */
  async ngOnInit(): Promise<void> {
    // fetch store
    this.objStoreInfo = await this.httpService.get('store/info/' + this.storeID)
    this.center = { lat: this.objStoreInfo.location.coordinates[1], lng: this.objStoreInfo.location.coordinates[0] };
    // set map location
    this.markerPositions.push({ lat: this.objStoreInfo.location.coordinates[1], lng: this.objStoreInfo.location.coordinates[0] });
  
    // get filters for make & model
    const filters = (await this.httpService.get('filter') as any).data;
    this.arrMake = filters.find((ele: any) => ele.name.value === 'make').data;
    this.arrModel = filters.find((ele: any) => ele.name.value === 'model').data;
    this.arrModelCopy = JSON.parse(JSON.stringify(this.arrModel));
  
    //disbale model field default
    this.formFilter.get('model')?.disable();
  
    // fetch vehicles
    await this.getStoreVehicles();
  }


  /**
   * Fetches vehicles for the current store based on pagination parameters.
   * 
   * This method retrieves a list of cars associated with the current store,
   * applying pagination if specified. It updates the component's car list
   * and pagination state.
   * 
   * @param event - Optional PaginatorState object containing pagination information.
   *                If provided, it updates the current page and limit values.
   * 
   * @returns A Promise that resolves when the car data has been fetched and stored.
   *          The fetched data is stored in the `arrCars` property of the component.
   */
  async getStoreVehicles(event?: PaginatorState): Promise<void> {
    if (event) {
      this.page = event.page ?? 0;
      this.limit = event.rows ?? 15;
    }
    this.arrCars = await this.httpService.get(`car/store/${this.objStoreInfo._id}?page=${this.page}&limit=${this.limit}`);
  }

  /**
   * Navigates to the car details page for a specific car.
   * 
   * This function uses the Angular Router to navigate to the car details page,
   * using the provided car ID to construct the URL.
   *
   * @param id - The unique identifier of the car to view details for.
   * @returns void This function doesn't return a value, but triggers a navigation action.
   */
  redirectToCarDetails(id: string): void {
    this.router.navigateByUrl(`/detail/${id}`);
  }

  /**
   * Finds the primary image of a car.
   * 
   * This function searches through the car's images to find the one with the lowest position value,
   * which is considered the primary image.
   *
   * @param car - The car object containing an array of images.
   * @param car.images - An array of image objects associated with the car.
   * @param car.images[].position - The position value of each image.
   * 
   * @returns The image object with the lowest position value, or undefined if no images are found.
   */
  findCarPrimaryImage(car: any) {
    return car.images.find((image: any) => image.position === Math.min(...car.images.map((img: any) => img.position)));
  }

  /**
   * Checks if a specific service is available for a car.
   * 
   * This function searches through an array of services to find a match for the specified service.
   * It returns the matching service object if found, or false if the service is not available
   * or if the services array is empty or undefined.
   *
   * @param {any[]} services - An array of service objects associated with the car.
   * @param {string} service - The name of the service to search for.
   * @returns {any | boolean} The matching service object if found, or false if not found or if services is empty/undefined.
   */
  findCarServices(services: any, service: string): any | boolean {
    return services && services.length > 0 ? services.find((srvc: any) => srvc.value === service) : false;
  }

  /**
   * Handles the change event when a make is selected in the filter form.
   * This function updates the available models based on the selected make and enables the model field.
   *
   * @param {any} event - The change event object from the make dropdown.
   *                      Expected to have a 'value' property containing the selected make.
   * 
   * @returns {void} This function doesn't return a value, but it updates the component's state:
   *                 - Updates the arrModel array with models related to the selected make.
   *                 - Enables the 'model' field in the filter form.
   */
  onChangeMake(event: any): void {
    // find related models to make and set
    const dependantData = this.arrModelCopy.filter((element: any) => element.join === event.value);
    this.arrModel = dependantData;
  
    // enable model field
    this.formFilter.get('model')?.enable();
  }

  /**
   * Submits the filter form and fetches filtered car results for the current store.
   * 
   * This function handles pagination and applies the current filter values when fetching cars.
   * It updates the page and limit values based on the provided event or resets them to default values.
   * The filtered results are then stored in the `arrCars` property.
   *
   * @param {PaginatorState} [event] - Optional paginator state object containing pagination information.
   *                                   If provided, it updates the current page and limit values.
   * 
   * @returns {Promise<void>} A promise that resolves when the filtered car data has been fetched and stored.
   *                          The fetched data is stored in the `arrCars` property of the component.
   */
  async submitFormFilter(event?: PaginatorState): Promise<void> {
    if (event) {
      this.page = event.page ?? 0;
      this.limit = event.rows ?? 15;
    } else {
      this.page = 0;
      this.limit = 15;
    }
    this.arrCars = await this.httpService.post(`car/client/storefilter?page=${this.page}&limit=${this.limit}`, { ...this.formFilter.value, store: this.objStoreInfo._id })
  }
}
