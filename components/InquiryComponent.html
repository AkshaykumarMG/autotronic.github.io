import { Component, OnInit, ViewEncapsulation } from '@angular/core';
import { environment } from '../../../../environments/environment';

// customn service
import { CSVService } from '../../../utilities/services/csv.service';
import { HttpService } from '../../../utilities/services/http.service';
import { LocalStorageService } from '../../../utilities/services/local-storage.service';

// primeng
import { MessageService } from 'primeng/api';
import { TablePageEvent } from 'primeng/table';

@Component({
  selector: 'app-inquiry',
  templateUrl: './inquiry.component.html',
  styleUrl: './inquiry.component.scss',
  encapsulation: ViewEncapsulation.None
})
export class InquiryComponent implements OnInit {

  arrInquiry: any;
  arrStatus = [
    { label: 'New', value: 'new', color: 'green', icon: 'assignment_add', count: 0 },
    { label: 'Pending', value: 'pending', color: 'orange', icon: 'pending_actions', count: 0 },
    { label: 'Done', value: 'done', color: 'red', icon: 'assignment_turned_in', count: 0 },
  ];
  strCurrentTabStatus: string = 'new';
  isEditDialogOpen: boolean = false;
  isViewDialogOpen: boolean = false;

  // update inquiry status
  currentRecordToEdit = { _id: '', status: '', note: '', user: this.localStorageService.getObject('user').identifier };
  currentRecordToView: any;

  // view store timeline dialog
  isInquiryTimelineDialogOpen: boolean = false;
  objInquiryTimelineDetail = { name: '', timeline: [] };

  constructor(
    private readonly csvService: CSVService,
    private readonly httpService: HttpService,
    private readonly messageService: MessageService,
    private readonly localStorageService: LocalStorageService
  ) { }

  /**
   * Initializes the component by fetching inquiry data for the current tab status.
   * This method is called automatically when the component is initialized.
   * 
   * @returns {Promise<void>} A promise that resolves when the initialization is complete.
   */
  async ngOnInit(): Promise<void> {
    await this.getInquiryByStore(this.strCurrentTabStatus);
  }

  /**
   * Filters the inquiry array based on the given status.
   * 
   * @param status - The status to filter inquiries by.
   * @returns An array of inquiry objects that match the specified status.
   */
  getInquiryByFilterStatus(status: string) {
    return this.arrInquiry.filter((inquiry: any) => inquiry.status === status);
  }

  /**
   * Extracts the make and model from a given vehicle name string.
   * Assumes the name is in the format "Year Make Model" (e.g., "2022 Toyota Camry").
   *
   * @param name - The full name of the vehicle, including year, make, and model.
   * @returns An object containing the extracted make and model.
   *          The 'make' property is the second word in the name string.
   *          The 'model' property is the third word in the name string.
   */
  getMakeModelByName(name: string) {
    const splitTitle = name.split(' ');
    return { make: splitTitle[1], model: splitTitle[2] };
  }

  /**
   * Fetches and updates inquiry data for a specific store based on the given status and pagination parameters.
   * 
   * This method retrieves inquiry data from the server, updates the component's inquiry array,
   * and refreshes the status counts. It supports pagination through the optional TablePageEvent parameter.
   *
   * @param status - The status of inquiries to fetch (e.g., 'new', 'pending', 'done').
   * @param event - Optional. A TablePageEvent object containing pagination information.
   *                If provided, it will be used to determine the page and limit for the data fetch.
   * 
   * @returns {Promise<void>} A promise that resolves when the inquiry data has been fetched and updated.
   */
  async getInquiryByStore(status: string, event?: TablePageEvent): Promise<void> {
    // set changed status
    this.strCurrentTabStatus = status;

    // set default value for pagination
    let page = 0;
    let limit = 10;

    // check if paginate then update default value
    if (event) {
      page = event.first;
      limit = event.rows;
    }

    // get stores
    const store = this.localStorageService.get('store');

    // fetch inquiries by store with pagination
    const resInquiry: any = await this.httpService.get(`inquiry/store/${store}/${status}?page=${page}&limit=${limit}`);
    this.arrInquiry = resInquiry.data;

    // update count of statuses
    this.arrStatus.forEach((element: any) => {
      const status = resInquiry.count.find((cnt: any) => cnt._id === element.value);
      element.count = status ? status.count : 0;
    });
  }

  /**
   * Retrieves the status label object based on the provided status value.
   * 
   * This function searches through the `arrStatus` array to find a status
   * object that matches the given status value.
   *
   * @param status - The status value to search for (e.g., 'new', 'pending', 'done').
   * @returns The matching status object if found, or undefined if no match is found.
   *          The returned object typically includes properties like label, value, color, and icon.
   */
  getStatusLabelByValue(status: string) {
    return this.arrStatus.find((element: any) => element.value === status);
  }

  /**
   * Opens the edit dialog for a record and sets the current record to edit.
   * 
   * This function prepares the edit dialog by setting the dialog state to open
   * and populating the currentRecordToEdit object with data from the selected record.
   * It also retrieves the current user's identifier from local storage.
   *
   * @param data - The data object of the record to be edited.
   *               Expected to have properties: _id, status, and note.
   * @returns {void}
   */
  openRecordEditDialog(data: any): void {
    this.isEditDialogOpen = true;
    this.currentRecordToEdit._id = data._id;
    this.currentRecordToEdit.status = data.status;
    this.currentRecordToEdit.note = data.note;
    this.currentRecordToEdit.user = this.localStorageService.getObject('user').identifier;
  }

  /**
   * Closes the record edit dialog and resets the current record being edited.
   * 
   * This function sets the edit dialog's open state to false and clears the
   * currentRecordToEdit object, resetting all its properties to empty strings.
   * This effectively closes the dialog and prepares it for the next edit operation.
   * 
   * @returns {void}
   */
  closeRecordEditDialog(): void {
    this.isEditDialogOpen = false;
    this.currentRecordToEdit = { _id: '', status: '', note: '', user: '' };
  }

  /**
   * Submits updated inquiry data to the server and refreshes the inquiry list.
   * 
   * This asynchronous function performs the following actions:
   * 1. Sends a PATCH request to update the inquiry status on the server.
   * 2. Refreshes the inquiry list for the current tab status.
   * 3. Closes the edit dialog.
   * 4. Displays a success message if the update is successful.
   * 
   * If an error occurs during the process, it displays an error message.
   * 
   * @returns {Promise<void>} A promise that resolves when the update process is complete.
   * @throws {Error} If there's an issue with the HTTP request or any other part of the process.
   */
  async submitUpdatedInquiryData(): Promise<void> {
    try {
      await this.httpService.patch(`inquiry/status/${this.currentRecordToEdit._id}`, this.currentRecordToEdit);
      await this.getInquiryByStore(this.strCurrentTabStatus);
      this.closeRecordEditDialog();
      this.messageService.add({ severity: 'secondary', summary: 'Updated', detail: 'Inquiry status updated!' });
    } catch (error: any) {
      this.messageService.add({ severity: 'error', summary: 'Failed', detail: error });
    }
  }

  /**
   * Opens a dialog to view the details of a specific record.
   * 
   * This function sets the view dialog's open state to true and assigns the provided data
   * to the currentRecordToView property, making it available for display in the dialog.
   *
   * @param data - The data object containing details of the record to be viewed.
   *               This can be any type of object containing relevant information for display.
   * @returns {void}
   */
  openViewDialog(data: any): void {
    this.isViewDialogOpen = true;
    this.currentRecordToView = data;
  }

  /**
   * Redirects the user to a car detail page on the front-end website.
   * 
   * This function opens a new browser tab or window with the URL of the car detail page,
   * constructed using the front-end URL from the environment configuration and the provided car ID.
   * 
   * @param id - The unique identifier of the car to be displayed on the detail page.
   * @returns {void} This function does not return a value.
   */
  redirectCarToFrontURL(id: string): void {
    window.open(`${environment.front_url}detail/${id}`, '_blank');
  }

  /**
   * Downloads the inquiry data as a CSV file.
   * 
   * This function prepares the inquiry data for CSV export by:
   * 1. Cleaning the 'note' and 'message' fields to remove line breaks.
   * 2. Defining the CSV header and corresponding field names.
   * 3. Calling the csvService to generate and download the CSV file.
   * 
   * The resulting CSV file will be named 'inquiry.csv' and contain columns for
   * Username, Phone, Email Address, Type, Status, Message, Note, and Date.
   * 
   * @returns {void} This function does not return a value.
   */
  downloadCSV(): void {
    this.arrInquiry.forEach((element: any) => {
      element.note = element.note ? element.note.replace(/(\r\n|\n|\r)/gm, " ") : '';
      element.message = element.message.replace(/(\r\n|\n|\r)/gm, " ");
    });
    const header = ['Username', 'Phone', 'Email Address', 'Type', 'Status', 'Message', 'Note', 'Date'];
    const field = ['username', 'phone', 'email', 'type', 'status', 'message', 'note', 'date'];
    this.csvService.downloadFile(this.arrInquiry, header, field, 'inquiry.csv');
  }

  /**
   * Opens a dialog displaying the timeline details for a specific inquiry.
   * 
   * This asynchronous function fetches the inquiry timeline data from the server,
   * opens a dialog to display this information, and updates the component's state
   * with the fetched data.
   *
   * @param data - An object containing the inquiry data. It must include an '_id' property
   *               which is used to fetch the specific inquiry's timeline.
   * @returns A Promise that resolves when the timeline data has been fetched and the dialog
   *          has been opened with the data populated.
   */
  async openInquiryTimelineDetail(data: any): Promise<void> {
    // fetch the inquiry timeline data
    const response: any = await this.httpService.get(`inquiry/timeline/${data._id}`);
    // open dialog and set data
    this.isInquiryTimelineDialogOpen = true;
    this.objInquiryTimelineDetail.name = response.username;
    this.objInquiryTimelineDetail.timeline = response.timeline;
  }
}
