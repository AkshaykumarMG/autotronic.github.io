import { Component, ViewEncapsulation, inject } from '@angular/core';
import { Router, RouterModule } from '@angular/router';
import { FormControl, FormGroup, ReactiveFormsModule, Validators } from '@angular/forms';
import { environment } from '../../../environments/environment';

// custom services
import { HttpService } from '../../utilities/services/http.service';
import { LocalStorageService } from '../../utilities/services/local-storage.service';

// firebase messaging
import { Messaging, getToken } from '@angular/fire/messaging';

// primeng
import { MessageService } from 'primeng/api';
import { PasswordModule } from 'primeng/password';
import { InputTextModule } from 'primeng/inputtext';
import { DividerModule } from 'primeng/divider';
import { ButtonModule } from 'primeng/button';
import { ToastModule } from 'primeng/toast';
import { AnimateOnScrollModule } from 'primeng/animateonscroll';

@Component({
  selector: 'app-login',
  standalone: true,
  imports: [ReactiveFormsModule, RouterModule, PasswordModule, InputTextModule, DividerModule, ButtonModule, ToastModule, AnimateOnScrollModule],
  providers: [MessageService],
  templateUrl: './login.component.html',
  styleUrl: './login.component.scss',
  encapsulation: ViewEncapsulation.None
})
export class LoginComponent {

  // login form
  formLogin = new FormGroup({
    email: new FormControl(null, [Validators.required, Validators.email]),
    password: new FormControl(null, [Validators.required, Validators.minLength(8)])
  });

  // firebase notification
  private messaging = inject(Messaging);

  constructor(
    private readonly router: Router,
    private readonly httpService: HttpService,
    private readonly messageService: MessageService,
    private readonly localStorageService: LocalStorageService,
  ) { }

  /**
   * Handles the login form submission process.
   * 
   * This asynchronous function performs the following tasks:
   * 1. Submits the login form data to the authentication service.
   * 2. If successful, resets the form and registers for push notifications.
   * 3. Stores the user data in local storage.
   * 4. Redirects the user to the appropriate page based on stored navigation state.
   * 
   * @async
   * @returns {Promise<void>} A promise that resolves when the login process is complete.
   * @throws Will throw an error if the login process fails, which is then displayed as a toast message.
   */
  async onLoginSubmit(): Promise<void> {
    try {
      const data = this.formLogin.value;
      const result: any = await this.httpService.post('authentication/client-signin', data);
      if (result.success) {
        // reset form
        this.formLogin.reset();

        // get token for notification
        navigator.serviceWorker
          .register("/assets/firebase-messaging-sw.js", {
            type: "module",
          })
          .then((serviceWorkerRegistration) => {
            getToken(this.messaging, {
              vapidKey: environment.vapidKey,
              serviceWorkerRegistration,
            }).then((token) => {
              // This is a good place to then store it on your database for each user
              this.httpService.patch('authentication/notification-token', { token });
              this.localStorageService.set('ne', 'true');
            });
          });

        // set user data in loca storage
        this.localStorageService.setObject('user', result.data);

        // check state is pending for page redirection
        const moveto = this.localStorageService.get('moveto');
        // if then redirecvt there and clear move to from storage
        if (moveto) {
          this.localStorageService.remove('moveto');
          this.router.navigateByUrl(moveto);
        } else {
          // redirect user to home of panel
          this.router.navigate(['user/home']);
        }
      }
    } catch (error: any) {
      this.messageService.add({ severity: 'error', summary: 'Error', detail: error });
    }
  }

  /**
   * Clears the login form by resetting all form controls to their initial values.
   * This function is typically called when the user wants to clear all entered data from the form.
   *
   * @returns {void} This function does not return a value.
   */
  onClearForm(): void {
    this.formLogin.reset();
  }
}
