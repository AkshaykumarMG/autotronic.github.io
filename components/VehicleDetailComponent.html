  /**
 * Initializes the component by fetching and setting up various data.
 * This method is called automatically when the component is initialized.
 * 
 * It performs the following tasks:
 * - Fetches car details from the server
 * - Sets up SEO information
 * - Retrieves and processes filter data
 * - Prepares media (images and videos) for display
 * - Calculates distance if user location is available
 * - Checks if the car is in user's favorites
 * - Sets up finance information if applicable
 * - Prepares appointment and inquiry forms
 * - Calculates fuel economy if data is available
 * - Increments the car view count
 * 
 * @returns {Promise<void>} A promise that resolves when all initialization tasks are complete
 */
  async ngOnInit(): Promise<void> {
    this.objCarDetails = await this.httpService.get(`car/client/${this.strCurrentID}`);
    if (this.objCarDetails.user) {
      this.objCarDetails.user.phone = String(this.objCarDetails.user.phone); // conver user phone to string for slice
    } else {
      this.objCarDetails.store.phone = String(this.objCarDetails.store.phone); // conver user phone to string for slice
    }

    // set SEO
    this.seoService.setSEOPage(this.objCarDetails.name, this.objCarDetails.description, `${this.document.location.origin}/`);

    // get filters from api
    this.arrFilter = (await this.httpService.get('filter') as any).data;

    // add image and videos
    this.arrMedia = this.objCarDetails.images;
    // if video then merge with images
    if (this.objCarDetails.videos && this.objCarDetails.videos.length > 0) {
      this.objCarDetails.videos.forEach((element: any) => {
        this.arrMedia!.push(element);
      })
    }

    // get dynamic filters
    const objKeys = Object.keys(this.objCarDetails);
    this.arrFilter.forEach((element: any) => {
      if (objKeys.includes(element.name.value)) {
        if (element.data && element.type != 'multi-select' && element.type != 'multi' && element.type != 'textbox-list') {
          const findValue = element.data.find((ele: any) => ele.value === this.objCarDetails[element.name.value]);
          if (findValue) {
            this.arrFilterList.push({ label: element.name.label, value: findValue.label, field: element.name.value });
          }
        }
      }
    });

    // reposive parameters for diffrent devices
    this.responsiveOptions = [
      {
        breakpoint: '1024px',
        numVisible: 5
      },
      {
        breakpoint: '768px',
        numVisible: 3
      },
      {
        breakpoint: '560px',
        numVisible: 1
      }
    ];

    // check user give location permission - if permission then count distance
    const location = this.localStorageService.getObject('location');
    if (location && isPlatformBrowser(this.platformId)) {
      // count distance by library
      this.objCarDetails.distance = Math.ceil(getPreciseDistance(
        { latitude: location.lat, longitude: location.lng },
        {
          latitude: this.objCarDetails.user ? this.objCarDetails.user.location.coordinates[1] : this.objCarDetails.store.location.coordinates[1],
          longitude: this.objCarDetails.user ? this.objCarDetails.user.location.coordinates[0] : this.objCarDetails.store.location.coordinates[0]
        }
        // { latitude: this.objCarDetails.store.location.coordinates[1], longitude: this.objCarDetails.store.location.coordinates[0] }
      ) / 1000) + ' KM';
    }

    // check it's user favourite advertisement
    const user = this.localStorageService.getObject('user');
    if (user) {
      if (this.objCarDetails.favourite && this.objCarDetails.favourite.length > 0) {
        this.isUserFavourite = this.objCarDetails.favourite.includes(user.identifier) ? true : false;
      }
    }

    // set finance condition if it's not private person listing
    if (!this.objCarDetails.user && this.objCarDetails.store.finance_apr) {
      // find tac as per state
      const csv: any = await this.httpService.getWithOutsideURLCSV('assets/csv/rate-table.csv');
      const taxState: any = this.csvService.CSVToJSON(csv).find((element: any) => element.JRSD_CD === this.objCarDetails.store.address.state).VPR1_HST;
      if (taxState) {
        const aprInterest: number = this.objCarDetails.taxes_fees.reduce((acc: number, element: any) => {
          acc += element.mode === '%' ? (this.objCarDetails.asking_price * Number(element.value)) / 100 : Number(element.value);
          return acc;
        }, 0);

        this.objTaxInfo = {
          taxes: this.objCarDetails.taxes_fees,
          asking_price: this.objCarDetails.asking_price,
          total_price: this.objCarDetails.asking_price,
          term: this.calculatorTermByYear(this.objCarDetails.year),
          frequency: ['Weekly', 'Biweekly', 'Monthly'],
          apr: aprInterest,
          down_payment: 0,
          hst: this.objCarDetails.taxes_fees.find((element: any) => element.type === 'HST').value,
          hst_total: (this.objCarDetails.asking_price * this.objCarDetails.taxes_fees.find((element: any) => element.type === 'HST').value) / 100,
          // price: (this.objTaxInfo.total_price / this.formPayment.value.term! / frequency)
        };
        this.calculatePaymentForFinance();
      }

      this.formPayment.patchValue({
        apr: this.objCarDetails.store.finance_apr
      });
    }

    // set store value in appoitment form
    this.formAppointment.patchValue({
      car: this.objCarDetails._id as string,
      store: this.objCarDetails.user ? this.objCarDetails.user._id as string : this.objCarDetails.store._id as string,
      postal_code: this.localStorageService.get('code')
    })

    // set store value in inquiry form
    this.formInquirey.patchValue({

  /**
   * Calculates the available payment term options based on the vehicle's manufacturing year.
   * 
   * This function determines the appropriate payment term options (in months) for a vehicle
   * based on its year of manufacture. Newer vehicles typically have more extended payment options.
   *
   * @param year - The manufacturing year of the vehicle.
   * @returns An array of numbers representing the available payment terms in months.
   *          The array is sorted in ascending order.
   */
  calculatorTermByYear(year: number): number[] {
    if (year >= 2013 && year <= 2015) {
      return [12, 24, 36, 48]
    } else if (year == 2016) {
      return [12, 24, 36, 48, 60]
    } else if (year == 2017) {
      return [12, 24, 36, 48, 60, 72]
    } else if (year == 2018) {
      return [12, 24, 36, 48, 60, 72, 84]
    } else if (year >= 2019) {
      return [12, 24, 36, 48, 60, 72, 84, 96]
    }
    return [12, 24, 36]
  }

  /**
 * Calculates the installment for finance based on the form payment data.
 * It includes down payment, interest, and tax calculations.
 * The function also updates the total price and the installment amount.
 */
  calculatePaymentForFinance(event?: any) {

    // set tax toggle value in payment form
    if (event) {
      this.formPayment.get('tax_include')?.setValue(event.checked ? true : false);
    }

    // get updated data from form payment
    const data = this.formPayment.value;

    // check down payment
    if (data.down_payment! > 0) {
      // deduct downpayment from net amount
      const netPrice = this.objCarDetails.asking_price - data.down_payment!;
      // check tax should be included or not
      if (data.tax_include) {
        // add tax by region
        this.objTaxInfo.total_price = netPrice + this.objTaxInfo.apr;
      } else {
        this.objTaxInfo.total_price = netPrice;
      }
    } else {
      this.formPayment.patchValue({
        down_payment: 0
      });
      // check tax should be included or not
      if (data.tax_include) {
        // add tax by region
        this.objTaxInfo.total_price = this.objTaxInfo.asking_price + this.objTaxInfo.apr;
      } else {
        this.objTaxInfo.total_price = this.objTaxInfo.asking_price;
      }
    }

    // get frequency of payment
    const frequency = data.frequency === 'Weekly' ? 4 : data.frequency === 'Biweekly' ? 2 : 1;
    // calculate final installment
    this.objTaxInfo.price = this.objTaxInfo.total_price / data.term! / frequency;
  }

 * Calculates the tax amount based on the given percentage.

 * @param percentage - The percentage of tax to be applied.
