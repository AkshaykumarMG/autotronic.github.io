import { Component, OnInit } from '@angular/core';
import { FormControl, FormGroup, ReactiveFormsModule, Validators } from '@angular/forms';
import { Router, RouterModule } from '@angular/router';

// custom service
import { HttpService } from '../../utilities/services/http.service';
import { LocalStorageService } from '../../utilities/services/local-storage.service';

// primeng
import { ButtonModule } from 'primeng/button';
import { DropdownModule } from 'primeng/dropdown';
import { InputTextModule } from 'primeng/inputtext';
import { TabViewModule } from 'primeng/tabview';
import { DividerModule } from 'primeng/divider';
import { AnimateOnScrollModule } from 'primeng/animateonscroll';

@Component({
  selector: 'app-home',
  standalone: true,
  imports: [ReactiveFormsModule, RouterModule, TabViewModule, AnimateOnScrollModule, InputTextModule, DropdownModule, ButtonModule, DividerModule],
  templateUrl: './home.component.html',
  styleUrl: './home.component.scss'
})
export class HomeComponent implements OnInit {

  // defined variables
  arrMake: any;
  arrModel: any;
  arrBodyType: any;
  arrModelCopy: any;

  // search vehicle form
  formSearchVehicle: FormGroup = new FormGroup({
    make: new FormControl<string | null>(null, [Validators.required]),
    model: new FormControl<string | null>(null, [Validators.required]),
    postal_code: new FormControl<string | null>(null, [Validators.required])
  });

  // total cars
  strTotalCars: string = '0';

  constructor(
    private readonly router: Router,
    private readonly httpService: HttpService,
    private readonly localStorageService: LocalStorageService,
  ) { }

  /**
   * Initializes the component by retrieving and setting the postal code from local storage,
   * and initializing necessary data for the component.
   * 
   * This method is called automatically when the component is initialized.
   * It checks for a stored postal code in local storage and sets it in the form if found.
   * Then it calls onInitValues to initialize other necessary data for the component.
   * 
   * @returns A promise that resolves when the initialization process is complete.
   */
  async ngOnInit(): Promise<void> {
      // check code already exist in local storage
      const code = this.localStorageService.get('code');
      if (code) {
        this.formSearchVehicle.patchValue({ postal_code: code });
      }
  
      // init necessaery data
      await this.onInitValues();
    }

  /**
   * Initializes the component's data by fetching necessary information from the server.
   * This method retrieves the total number of cars, filters for make and model,
   * and body type data. It also disables the model field in the search form by default.
   *
   * @async
   * @returns {Promise<void>} A promise that resolves when all data has been fetched and processed.
   */
  async onInitValues(): Promise<void> {
    // get total car
    this.strTotalCars = (await this.httpService.get('car/count') as any);

    // get filters for make & model
    const filters = (await this.httpService.get('filter') as any).data;
    this.arrMake = filters.find((ele: any) => ele.name.value === 'make').data.sort((a: any, b: any) => a.value > b.value ? 1 : -1);
    this.arrModelCopy = filters.find((ele: any) => ele.name.value === 'model').data;

    // get body type
    this.arrBodyType = filters.find((ele: any) => ele.name.value === 'body_type').data;

    //disbale model field default
    this.formSearchVehicle.get('model')?.disable();
  }

  
  /**
   * Handles the change event when a make is selected in the search form.
   * This function filters and sorts the models based on the selected make,
   * updates the available models, and enables the model field in the form.
   *
   * @param {any} event - The change event object from the make selection dropdown.
   *                      Expected to have a 'value' property containing the selected make.
   * @returns {void}
   */
  onChangeMake(event: any): void {
  
    // find related models to make and set
    const dependantData = this.arrModelCopy.filter((element: any) => element.join === event.value);
    this.arrModel = dependantData.sort((a: any, b: any) => a.value > b.value ? 1 : -1);;
  
    // enable model field
    this.formSearchVehicle.get('model')?.enable();
  }

  /**
   * Submits the search form and navigates to the search page with the form data as query parameters.
   * 
   * This asynchronous function retrieves the current values from the search form,
   * and uses them to navigate to the '/search' route. The form values for make, model,
   * and postal code are passed as query parameters in the URL.
   * 
   * @async
   * @returns {Promise<void>} A promise that resolves when the navigation is complete.
   */
  async submitSearchForm(): Promise<void> {
    const data = this.formSearchVehicle.value;
    this.router.navigate(['/search'], { queryParams: { make: data.make, model: data.model, code: data.postal_code } });
  }
}
