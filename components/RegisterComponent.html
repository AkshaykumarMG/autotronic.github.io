import { Component, ViewEncapsulation } from '@angular/core';
import { Router, RouterModule } from '@angular/router';
import { FormControl, FormsModule, ReactiveFormsModule, UntypedFormGroup, Validators } from '@angular/forms';

// custom service
import { HttpService } from '../../utilities/services/http.service';

// primeng
import { MessageService } from 'primeng/api';
import { PasswordModule } from 'primeng/password';
import { InputTextModule } from 'primeng/inputtext';
import { DividerModule } from 'primeng/divider';
import { SelectButtonModule } from 'primeng/selectbutton';
import { ButtonModule } from 'primeng/button';
import { DialogModule } from 'primeng/dialog';
import { InputOtpModule } from 'primeng/inputotp';
import { CommonModule } from '@angular/common';
import { ToastModule } from 'primeng/toast';
import { InputMaskModule } from 'primeng/inputmask';
import { AnimateOnScrollModule } from 'primeng/animateonscroll';

@Component({
  selector: 'app-register',
  standalone: true,
  imports: [ReactiveFormsModule, FormsModule, CommonModule, RouterModule, AnimateOnScrollModule, PasswordModule, InputTextModule, InputMaskModule, DividerModule, SelectButtonModule, ToastModule, ButtonModule, DialogModule, InputOtpModule],
  providers: [MessageService],
  templateUrl: './register.component.html',
  styleUrl: './register.component.scss',
  encapsulation: ViewEncapsulation.None
})
export class RegisterComponent {

  strSignUpTitle = "Sign Up";
  isOTPSent = false;
  strOTP: string | undefined;
  strOTPError: boolean = false;
  isOTPValid: boolean = false;
  isRegisterLoading: boolean = false;

  UserRoleOptions: any[] = [
    { label: 'Dealer', value: 'dealer' },
    { label: 'Private Person', value: 'person' }
  ];

  // login form
  formRegister = new UntypedFormGroup({
    email: new FormControl(null, [Validators.required, Validators.email]),
  });

  constructor(
    private readonly router: Router,
    private readonly httpService: HttpService,
    private readonly messageService: MessageService
  ) { }

  /**
 * Closes the OTP pop-up and resets the OTP-related variables.
 *
 * This function is called when the user closes the OTP pop-up. It resets the OTP value,
 * the OTP sent flag, and the OTP error flag to their initial states.
 *
 * @returns {void} - This function does not return any value.
 */
  onCloseOTPPopUp(): void {
    this.strOTP = undefined;
    this.isOTPSent = false;
    this.strOTPError = false;
  }

  /**
 * Sends an OTP to the user's email for verification.
 *
 * This function is responsible for sending an OTP to the user's email address for account verification.
 * It retrieves the form data, sends a POST request to the server, and updates the UI based on the response.
 *
 * @remarks
 * The function performs the following steps:
 * 1. Sets the `isRegisterLoading` flag to true to indicate that the OTP sending process is in progress.
 * 2. Retrieves the form data.
 * 3. Sends a POST request to the server with the email data.
 * 4. If the OTP sending is successful, updates the `isOTPSent` flag to true.
 * 5. If an error occurs during the OTP sending process, displays an error message using the `messageService`.
 * 6. Sets the `isRegisterLoading` flag to false to indicate that the OTP sending process is complete.
 *
 * @returns {Promise<void>} - A promise that resolves when the OTP sending process is complete.
 */
  async OnSentOTPEmail(): Promise<void> {
    try {
      this.isRegisterLoading = true;
      const data = this.formRegister.value;
      const response: any = await this.httpService.post('authentication/client-signup-otp', data);
      if (response?.success) {
        this.isOTPSent = true;
      }
    } catch (error: any) {
      this.messageService.add({ severity: 'error', summary: 'Error', detail: error });
    } finally {
      this.isRegisterLoading = false;
    }
  }

  /**
 * Handles the verification of the OTP sent to the user's email.
 * This function sends a POST request to the server to verify the OTP.
 * If the OTP is valid, it updates the form controls and changes the sign-up title.
 * If the OTP is invalid, it sets the `strOTPError` flag to true.
 *
 * @remarks
 * This function is called when the user submits the OTP received via email.
 *
 * @returns {Promise<void>} - A promise that resolves when the OTP verification process is complete.
 */
  async OnVerifyOTP(): Promise<void> {

    // check otp is entered
    if (this.strOTPError) {
      this.strOTPError = false;
    }

    // start loading
    this.isRegisterLoading = true;

    // get data to send email
    const data = this.formRegister.value;
    try {
      // send request to verify
      const response: any = await this.httpService.post('authentication/verify-client-signup', { email: data.email, otp: this.strOTP?.trim() });
      // if success
      if (response?.success) {
        this.isOTPSent = false;
        this.isOTPValid = true;
        this.formRegister.addControl('otp', new FormControl(this.strOTP));
        this.formRegister.addControl('username', new FormControl(null, [Validators.required]));
        this.formRegister.addControl('role', new FormControl(null, [Validators.required]));
        this.formRegister.addControl('password', new FormControl(null, [Validators.required, Validators.minLength(10)]));
        this.formRegister.addControl('phone', new FormControl(null, [Validators.required]));
        this.strSignUpTitle = "Create Account";
      }
    } catch (error: any) {
      this.strOTPError = true;
    } finally {
      this.isRegisterLoading = false;
    }
  }

  /**
 * Handles the registration process by sending a POST request to the server.
 * This function is called when the user submits the registration form.
 *
 * @remarks
 * The function performs the following steps:
 * 1. Sets the `isRegisterLoading` flag to true to indicate that the registration process is in progress.
 * 2. Retrieves the form data.
 * 3. Removes any hyphens from the phone number.
 * 4. Sends a POST request to the server with the registration data.
 * 5. If the registration is successful, displays a success message and navigates to the login page.
 * 6. If an error occurs during the registration process, displays an error message and clears the phone number field.
 * 7. Sets the `isRegisterLoading` flag to false to indicate that the registration process is complete.
 *
 * @returns {Promise<void>} - A promise that resolves when the registration process is complete.
 */
  async onRegisterSubmit(): Promise<void> {
    this.isRegisterLoading = true;
    const data = this.formRegister.value;
    try {
      data.phone = Number(data.phone.replace(/-/g, ""));
      const result: any = await this.httpService.post('authentication/client-signup', data);
      if (result?.success) {
        this.messageService.add({ severity: 'success', summary: 'Success', detail: 'Your account successfully created, kindly login now.' });
        this.router.navigateByUrl('/login');
      }
    } catch (error: any) {
      this.formRegister.patchValue({
        phone: null
      })
      this.messageService.add({ severity: 'error', summary: 'Error', detail: error });
    } finally {
      this.isRegisterLoading = false;
    }
  }
}
