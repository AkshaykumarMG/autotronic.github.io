  /**
 * Initializes the component by loading city and province data, and setting up the finance application form.
 * 
 * This method performs the following tasks:
 * 1. Loads city and province data from a CSV file.
 * 2. Checks for existing finance applications in the database.
 * 3. If an application exists, it populates the form with the existing data.
 * 4. If no application exists, it adds a new applicant form.
 * 
 * @async
 * @returns {Promise<void>} A promise that resolves when the initialization is complete.
 */
  async ngOnInit(): Promise<void> {
    // find city and province from static CSV file
    const csv: any = await this.httpService.getWithOutsideURLCSV('assets/csv/canadacities.csv');
    const cities = this.commonService.CSVToJSON(csv).map((cities: any) => { return cities.city_ascii });
    this.data_city = [...new Set(cities)].map((cities: any) => { return { label: cities, value: cities.toLowerCase() } });
    const province = this.commonService.CSVToJSON(csv).map((cities: any) => { return cities.province_id });
    this.data_province = [...new Set(province)].map((cities: any) => { return { label: cities, value: cities.toLowerCase() } });

    // check any finance application found in database
    const application: any = await this.httpService.get('finance-application');
    if (application) {
      // set current form id for update
      this.strUpdateID = application._id;
      // set value of button
      this.strAddOrUpdateButtonTitle = 'Update Finance Applicantion';

      // set value in form
      const arrFinance = this.formFinance.get("data") as FormArray;

      // set value in form from application data
      application.data.forEach((applicant: any) => {
        // get form group
        const group = this.getApplicant();

        // set value in form group
        Object.keys(group.controls).forEach(key => {

          // if previous employer 
          if (key === 'previous_employers' && applicant[key] && applicant[key].length > 0) {
            // set previous employer in form
            const arrPreviousEmployer = group.get("previous_employers") as FormArray;

            // set previous employer in form
            applicant[key].forEach((previousEmployer: any, index: number) => {

              // get previous employer form group
              const groupPreviousEmployer = this.getPreviousEmployer(group);

              // set value in form group
              Object.keys(previousEmployer).forEach(key => {
                groupPreviousEmployer.patchValue({
                  [key]: previousEmployer[key] ?? null
                });
              });

              // add in previous employer from array to form
              arrPreviousEmployer.push(groupPreviousEmployer)
            });
            return
          }

          // add value by key
          group.patchValue({
            [key]: applicant[key] ?? null
          });
        });

        // add in data from array
        arrFinance.push(group);
      });
    } else {
      this.addApplicant();
    }
  }

  /**
   * Retrieves the finance data FormArray from either a provided FormGroup or the main form.
   * 
   * This function is used to access the 'data' FormArray, which contains the finance application information.
   * If an element is provided, it retrieves the 'data' FormArray from that element. Otherwise, it retrieves
   * the 'data' FormArray from the main formFinance FormGroup.
   *
   * @param {FormGroup} [element] - Optional FormGroup from which to retrieve the 'data' FormArray.
   * @returns {FormArray} The 'data' FormArray containing finance application information.
   */
  getFinanceData(element?: FormGroup): FormArray {
    if (element) {
      return element.get("data") as FormArray;
    }
    return this.formFinance.get("data") as FormArray;
  }

  /**

   * Toggles the previous employer duration status based on the current employment duration.
   * If the total duration is 15 months or less, it adds a previous employer and sets the duration status to false.
   * If the duration is more than 15 months, it sets the duration status to true and clears any existing previous employers.
   *
   * @param {FormGroup} element - The form group containing the employment duration controls and previous employer information.
   * @returns {void} This function doesn't return a value; it modifies the form state directly.
   */
  togglePreviousEmployerDurationStatus(element: FormGroup): void {
    const year = element.get("current_employer_duration_year") as FormControl;
    const month = element.get("current_employer_duration_month") as FormControl;
    const duration = year.value == 0 ? 0 + month.value : (year.value * 12) + month.value;
    const duration_status = element.get("duration_status") as FormControl;
    if (duration <= 15) {
      this.addPreviousEmployer(element)
      duration_status.setValue(false);
    } else {
      duration_status.setValue(true);
      (element.get("previous_employers") as FormArray).clear();
    }
  }

  /**
   * Retrieves the FormArray of previous employers from a given FormGroup.
   * 
   * This function accesses the 'previous_employers' control within the provided FormGroup,
   * which is expected to be a FormArray containing information about an applicant's previous employers.
   *
   * @param {FormGroup} element - The FormGroup containing the 'previous_employers' FormArray.
   *                              This is typically the form group representing an individual applicant.
   * 
   * @returns {FormArray} A FormArray containing FormGroups, each representing a previous employer's details.
   *                      If the 'previous_employers' control doesn't exist or is not a FormArray, 
   *                      this may return undefined or throw an error.
   */
  getPreviousEmployerData(element: FormGroup): FormArray {
    return element.get("previous_employers") as FormArray;
  }

  /**
   * Removes a previous employer entry from the applicant's form.
   * 
   * This function removes a specific previous employer entry from the FormArray
   * of previous employers within the given applicant's FormGroup.
   *
   * @param {FormGroup} element - The FormGroup representing the applicant's data,
   *                              containing the 'previous_employers' FormArray.
   * @param {number} index - The index of the previous employer entry to be removed
   *                         from the 'previous_employers' FormArray.
   * @returns {void} This function does not return a value. It directly modifies
   *                 the form structure by removing an element from the FormArray.
   */
  removePreviousEmployer(element: FormGroup, index: number): void {
    this.getPreviousEmployerData(element).removeAt(index)
  }

  /**
 * Manages the finance application by either adding a new application or updating an existing one.
 * 
 * This function processes the finance application data from the form, filters out null values,
 * and either adds a new application to the database or updates an existing one based on the 'title' parameter.
 * It also handles success and error messages using a message service.
 *
 * @param {string} title - The title of the action, either 'Add Finance Applicantion' or 'Update Finance Applicantion'.
 * @returns {Promise<void>} A promise that resolves when the operation is complete.
 * @throws {Error} Throws an error if the HTTP request fails, which is caught and displayed as an error message.
 */
  async manageFinanceApplication(title: string): Promise<void> {

    // get finance application data from form
    const data = this.formFinance.value.data;

    // filter out null key value from object and map to array of objects with key-value pairs only
    const tempResult = data?.map((applicant: any) => {
      // remove previous employers if it's empty
      if (applicant.previous_employers && applicant.previous_employers.length === 0) {
        delete applicant.previous_employers;
      }

      const tempObject: any = {};
      // remove null key value from object
      Object.keys(applicant).filter((k) => applicant[k] !== null || undefined).forEach((k) => { tempObject[k] = applicant[k] });
      return tempObject;
    });

    if (title === 'Add Finance Applicantion') {
      // get user identifier from local storage
      const user = this.localStorageService.getObject('user').identifier;

      // add in database
      await this.httpService.post(`finance-application`, { user, data: tempResult }).then((response: any) => {
        if (response.success) {
          this.messageService.add({ severity: 'secondary', summary: 'Success', detail: 'Finance Application Added Successfully.' });
          this.strAddOrUpdateButtonTitle = 'Update Finance Applicantion';
          this.strUpdateID = response.id;
        }
      }).catch((error: string) => {
        this.messageService.add({ severity: 'error', summary: 'Error', detail: error });
      });
    } else {
      // update in database
      await this.httpService.patch(`finance-application/${this.strUpdateID}`, { data: tempResult }).then((response: any) => {
        if (response.success) {
          this.messageService.add({ severity: 'secondary', summary: 'Success', detail: 'Finance Application Updated Successfully.' });
        }
      }).catch((error: string) => {
        this.messageService.add({ severity: 'error', summary: 'Error', detail: error });
      });
    }
  }
}
